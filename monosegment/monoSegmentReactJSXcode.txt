// Lightweight Binary Parser to extract true DPI from PNG/JPEG headers
const extractDPI = (dataUrl) => {
  try {
    const marker = dataUrl.split(',')[0];
    const b64Part = dataUrl.split(',')[1].substring(0, 4000); 
    const bstr = atob(b64Part);
    
    if (marker.includes('image/png')) {
      let offset = 8;
      while (offset < bstr.length - 12) {
        const length = (bstr.charCodeAt(offset) << 24) | (bstr.charCodeAt(offset+1) << 16) | (bstr.charCodeAt(offset+2) << 8) | bstr.charCodeAt(offset+3);
        const type = bstr.substring(offset+4, offset+8);
        if (type === 'pHYs') {
          const ppuX = (bstr.charCodeAt(offset+8) << 24) | (bstr.charCodeAt(offset+9) << 16) | (bstr.charCodeAt(offset+10) << 8) | bstr.charCodeAt(offset+11);
          const unit = bstr.charCodeAt(offset+16);
          if (unit === 1) return Math.round(ppuX * 0.0254); 
        }
        if (type === 'IDAT') break;
        offset += 12 + length;
      }
    } else if (marker.includes('image/jpeg')) {
      if (bstr.charCodeAt(0) === 0xFF && bstr.charCodeAt(1) === 0xD8) {
        let offset = 2;
        while (offset < bstr.length - 4) {
          if (bstr.charCodeAt(offset) !== 0xFF) break;
          const markerType = bstr.charCodeAt(offset+1);
          if (markerType === 0xE0) { 
            if (bstr.substring(offset+4, offset+8) === 'JFIF') {
              const units = bstr.charCodeAt(offset+11);
              const xDensity = (bstr.charCodeAt(offset+12) << 8) | bstr.charCodeAt(offset+13);
              if (units === 1) return xDensity; 
              if (units === 2) return Math.round(xDensity * 2.54); 
            }
          }
          const length = (bstr.charCodeAt(offset+2) << 8) | bstr.charCodeAt(offset+3);
          offset += 2 + length;
        }
      }
    }
  } catch (e) { 
    console.error('DPI Extraction Error', e); 
  }
  return 72; 
};

// --- CORE VECTOR PATH BUILDER ---
const buildPath2D = (path) => {
  const p = new Path2D();
  if (path.type === 'auto') {
    p.moveTo(path.p1.x, path.p1.y);
    if (path.cp1) p.bezierCurveTo(path.cp1.x, path.cp1.y, path.cp2.x, path.cp2.y, path.p2.x, path.p2.y);
    else p.lineTo(path.p2.x, path.p2.y);
  } else if (path.type === 'pen' && path.nodes && path.nodes.length > 0) {
    p.moveTo(path.nodes[0].x, path.nodes[0].y);
    for (let i = 1; i < path.nodes.length; i++) {
      p.bezierCurveTo(path.nodes[i-1].handleOut.x, path.nodes[i-1].handleOut.y, path.nodes[i].handleIn.x, path.nodes[i].handleIn.y, path.nodes[i].x, path.nodes[i].y);
    }
  } else if (path.type === 'brush' && path.points && path.points.length > 0) {
    p.moveTo(path.points[0].x, path.points[0].y);
    if (path.points.length > 2 && path.smoothness > 0) {
      for (let i = 2; i < path.points.length; i++) {
        const p2 = path.points[i - 1];
        const p3 = path.points[i];
        const mid = { x: (p2.x + p3.x) / 2, y: (p2.y + p3.y) / 2 };
        p.quadraticCurveTo(p2.x, p2.y, mid.x, mid.y);
      }
      p.lineTo(path.points[path.points.length - 1].x, path.points[path.points.length - 1].y);
    } else {
      for (let i = 1; i < path.points.length; i++) {
        p.lineTo(path.points[i].x, path.points[i].y);
      }
    }
  }
  return p;
};

// --- RENDER ENGINE ---
const renderAllPaths = (ctx, w, h, pathsToRender) => {
  ctx.clearRect(0, 0, w, h);
  pathsToRender.forEach(p => {
    ctx.globalCompositeOperation = p.isEraser ? 'destination-out' : 'source-over';
    ctx.strokeStyle = p.color;
    ctx.fillStyle = p.color;
    ctx.lineWidth = p.thickness;
    ctx.lineCap = 'round';
    ctx.lineJoin = 'round';

    const p2d = buildPath2D(p);
    ctx.stroke(p2d);

    if (p.type === 'brush' && p.points.length === 1) {
       ctx.beginPath();
       ctx.arc(p.points[0].x, p.points[0].y, p.thickness / 2, 0, Math.PI * 2);
       ctx.fill();
    }
  });
  ctx.globalCompositeOperation = 'source-over';
};

// --- LETTER PLACER ALGORITHM (Python Logic -> JS Native) ---
const placeLetters = async (maskCanvas, imgObj, settings) => {
  // Force font load explicitly to guarantee rendering before measuring
  try {
    const primaryFont = settings.font.split(',')[0].replace(/['"]/g, '').trim();
    await document.fonts.load(`bold 20px "${primaryFont}"`);
  } catch (err) { console.warn('Font load trigger failed'); }
  await document.fonts.ready;

  const { w, h } = imgObj;
  const lettersCanvas = document.createElement('canvas');
  lettersCanvas.width = w; lettersCanvas.height = h;
  const ctxL = lettersCanvas.getContext('2d');

  const combinedCanvas = document.createElement('canvas');
  combinedCanvas.width = w; combinedCanvas.height = h;
  const ctxC = combinedCanvas.getContext('2d', { willReadFrequently: true });
  
  const baseImg = new Image();
  baseImg.src = imgObj.dataUrl;
  await new Promise(r => { baseImg.onload = r; });
  
  ctxC.drawImage(baseImg, 0, 0);
  const baseData = ctxC.getImageData(0, 0, w, h).data;
  
  // Read exact mask to prevent colored lines (like red) from breaking the flood fill
  const maskData = maskCanvas.getContext('2d').getImageData(0, 0, w, h).data;
  
  const labels = new Int32Array(w * h);
  const regions = [];
  
  // Flood Fill looking strictly for WHITE/EMPTY areas
  let currentLabel = 1;
  for (let y = 0; y < h; y += 2) { 
      for (let x = 0; x < w; x += 2) {
          const startIdx = y * w + x;
          if (labels[startIdx] === 0) {
              const r = baseData[startIdx * 4];
              const g = baseData[startIdx * 4 + 1];
              const b = baseData[startIdx * 4 + 2];
              const a = baseData[startIdx * 4 + 3];

              const isMaskWall = maskData[startIdx * 4 + 3] > 128;
              const isBaseWall = (r < 200 && g < 200 && b < 200) || (a <= 128);

              if (!isMaskWall && !isBaseWall) {
                  const q = [startIdx];
                  labels[startIdx] = currentLabel;
                  let count = 0;
                  let sumX = 0, sumY = 0;
                  let touchesBorder = false;
                  
                  let head = 0;
                  while(head < q.length) {
                      const curr = q[head++];
                      const cx = curr % w;
                      const cy = Math.floor(curr / w);
                      
                      count++;
                      sumX += cx; sumY += cy;
                      
                      if (cx <= 0 || cx >= w-1 || cy <= 0 || cy >= h-1) touchesBorder = true;
                      
                      const neighbors = [curr - 1, curr + 1, curr - w, curr + w];
                      for(let i=0; i<4; i++) {
                          const n = neighbors[i];
                          if (n >= 0 && n < labels.length && labels[n] === 0) {
                              const nr = baseData[n*4];
                              const ng = baseData[n*4+1];
                              const nb = baseData[n*4+2];
                              const na = baseData[n*4+3];

                              const nIsMaskWall = maskData[n*4 + 3] > 128;
                              const nIsBaseWall = (nr < 200 && ng < 200 && nb < 200) || (na <= 128);

                              if (!nIsMaskWall && !nIsBaseWall) {
                                  labels[n] = currentLabel;
                                  q.push(n);
                              } else {
                                  labels[n] = -1; // Wall
                              }
                          }
                      }
                  }
                  
                  if (!touchesBorder && count > 100) {
                      regions.push({
                          id: currentLabel,
                          cx: sumX / count, // Center of Mass (Centroid)
                          cy: sumY / count
                      });
                  }
                  currentLabel++;
              } else {
                  labels[startIdx] = -1; // Wall
              }
          }
      }
  }

  // Pre-calculate Spiral Nudge Offsets
  const step = 3;
  const maxNudge = 300;
  const offsets = [];
  for(let dy = -maxNudge; dy <= maxNudge; dy += step) {
      for(let dx = -maxNudge; dx <= maxNudge; dx += step) {
          offsets.push({ dx, dy, dist: dx*dx + dy*dy });
      }
  }
  offsets.sort((a,b) => a.dist - b.dist);
  
  // Test Canvas for pixel-perfect letter measurement
  const testCanvas = document.createElement('canvas');
  const testCtx = testCanvas.getContext('2d', { willReadFrequently: true });
  
  const safetyPx = Number(settings.safetyPx) || 0;
  let defaultLow = Number(settings.minSize) || 4;
  let defaultHigh = Number(settings.maxSize) || 35;
  if (defaultLow > defaultHigh) { const t = defaultLow; defaultLow = defaultHigh; defaultHigh = t; }

  for(let r of regions) {
      
      // Abstracted the binary search to support Python's 2-Pass safety fallback
      const tryFit = (margin) => {
          let best = null;
          let low = defaultLow, high = defaultHigh;
          const padding = (margin * 2) + 10;
          
          while(low <= high) {
              let mid = Math.floor((low+high)/2);
              
              testCtx.font = `bold ${mid}px ${settings.font}`;
              testCtx.textBaseline = 'middle';
              testCtx.textAlign = 'center';
              const metrics = testCtx.measureText(settings.char);
              
              // Fallback for browsers that don't fully support actualBoundingBox
              const bw = metrics.actualBoundingBoxRight ? (metrics.actualBoundingBoxRight + metrics.actualBoundingBoxLeft) : testCtx.measureText(settings.char).width;
              const bh = metrics.actualBoundingBoxAscent ? (metrics.actualBoundingBoxAscent + metrics.actualBoundingBoxDescent) : mid;
              
              const fontW = Math.ceil(bw + padding);
              const fontH = Math.ceil(bh + padding);
              
              if (fontW <= 0 || fontH <= 0 || isNaN(fontW) || isNaN(fontH)) break;
              
              testCanvas.width = fontW; testCanvas.height = fontH;
              testCtx.clearRect(0,0,fontW,fontH);
              // Important: Re-assign font properties after width/height change clears them
              testCtx.font = `bold ${mid}px ${settings.font}`;
              testCtx.textBaseline = 'middle';
              testCtx.textAlign = 'center';
              
              // Draw fat stroke for safety bounds
              testCtx.lineJoin = 'round';
              if (margin > 0) {
                  testCtx.lineWidth = margin * 2;
                  testCtx.strokeStyle = 'black';
                  testCtx.strokeText(settings.char, fontW/2, fontH/2);
              }
              testCtx.fillStyle = 'black';
              testCtx.fillText(settings.char, fontW/2, fontH/2);
              
              const lData = testCtx.getImageData(0,0,fontW,fontH).data;
              const letterPixels = [];
              
              // Step by 1 to ensure tiny fonts aren't missed
              for(let ly=0; ly<fontH; ly+=1) {
                  for(let lx=0; lx<fontW; lx+=1) {
                      const idx = (ly * fontW + lx) * 4;
                      if (lData[idx+3] > 128) {
                          letterPixels.push({ x: lx - Math.floor(fontW/2), y: ly - Math.floor(fontH/2) });
                      }
                  }
              }
              
              let foundPos = null;
              for(let o of offsets) {
                  const testX = Math.round(r.cx + o.dx);
                  const testY = Math.round(r.cy + o.dy);
                  
                  if (testX - Math.floor(fontW/2) < 0 || testX + Math.floor(fontW/2) >= w || testY - Math.floor(fontH/2) < 0 || testY + Math.floor(fontH/2) >= h) continue;

                  let valid = true;
                  for(let i=0; i<letterPixels.length; i++) {
                      const px = testX + letterPixels[i].x;
                      const py = testY + letterPixels[i].y;
                      if (labels[py * w + px] !== r.id) {
                          valid = false;
                          break;
                      }
                  }
                  if (valid) { foundPos = { x: testX, y: testY }; break; }
              }
              
              if (foundPos) {
                  best = { size: mid, x: foundPos.x, y: foundPos.y };
                  low = mid + 1; // It fits! Push for a larger size.
              } else {
                  high = mid - 1; // Failed. Drop the size.
              }
          }
          return best;
      };

      // PASS 1: Try requested safety margin
      let bestFit = tryFit(safetyPx);
      
      // PASS 2 (Python Logic): If it fails, fallback to MIN_SAFETY_PX = 2
      if (!bestFit && safetyPx > 2) {
          bestFit = tryFit(2);
      }
      
      // Stamp the absolute best fit
      if (bestFit) {
          ctxL.font = `bold ${bestFit.size}px ${settings.font}`;
          ctxL.fillStyle = settings.color;
          ctxL.textBaseline = 'middle';
          ctxL.textAlign = 'center';
          ctxL.fillText(settings.char, bestFit.x, bestFit.y);
      }
  }
  
  return lettersCanvas;
};

// --- CRC32 helper for PNG chunk integrity ---
const crc32 = (data) => {
  let crc = 0xFFFFFFFF;
  for (let i = 0; i < data.length; i++) {
    crc ^= data[i];
    for (let j = 0; j < 8; j++) {
      crc = (crc & 1) ? (0xEDB88320 ^ (crc >>> 1)) : (crc >>> 1);
    }
  }
  return (crc ^ 0xFFFFFFFF) >>> 0;
};

// --- Injects pHYs DPI metadata chunk into a PNG blob ---
const injectPngDpi = async (blob, dpi) => {
  const buffer = await blob.arrayBuffer();
  const bytes = new Uint8Array(buffer);
  const ppm = Math.round(dpi / 0.0254);
  const phys = new Uint8Array(21);
  phys[0]=0; phys[1]=0; phys[2]=0; phys[3]=9;
  phys[4]=112; phys[5]=72; phys[6]=89; phys[7]=115;
  phys[8]=(ppm>>24)&255; phys[9]=(ppm>>16)&255; phys[10]=(ppm>>8)&255; phys[11]=ppm&255;
  phys[12]=(ppm>>24)&255; phys[13]=(ppm>>16)&255; phys[14]=(ppm>>8)&255; phys[15]=ppm&255;
  phys[16]=1;
  const crcData = phys.slice(4, 17);
  const crc = crc32(crcData);
  phys[17]=(crc>>24)&255; phys[18]=(crc>>16)&255; phys[19]=(crc>>8)&255; phys[20]=crc&255;
  const result = new Uint8Array(bytes.length + 21);
  result.set(bytes.slice(0, 33), 0);
  result.set(phys, 33);
  result.set(bytes.slice(33), 54);
  return new Blob([result], { type: 'image/png' });
};

  // --- 🧠 TOPOLOGICALLY AWARE PROCESSING CORE ---
  const runAwareSegmenterCore = async (imgObj, params) => {
    const startTime = performance.now();
    await new Promise(resolve => setTimeout(resolve, 50)); 

    const thick = Number(params.thick) || 1;
    const minCut = Number(params.minCut) || 50;
    const maxCut = Number(params.maxCut) || 200;
    const spacing = Number(params.spacing) || 100;
    const smoothness = Number(params.smoothness) || 0;
    const color = params.color || '#000000';
    
    const w = imgObj.w; 
    const h = imgObj.h;
    
    const img = new Image();
    img.src = imgObj.dataUrl;
    await new Promise((resolve) => { img.onload = resolve; });

    const canvas = document.createElement('canvas');
    canvas.width = w; canvas.height = h;
    const ctx = canvas.getContext('2d', { willReadFrequently: true });
    ctx.drawImage(img, 0, 0);

    const maskCanvas = document.createElement('canvas');
    maskCanvas.width = w; maskCanvas.height = h;
    const maskCtx = maskCanvas.getContext('2d');

    const data = ctx.getImageData(0, 0, w, h).data;

    const isWall = (x, y) => {
      x = Math.floor(x); y = Math.floor(y);
      if (x < 0 || x >= w || y < 0 || y >= h) return true;
      const idx = (y * w + x) * 4;
      if (data[idx + 3] < 128) return true; 
      const brightness = (data[idx] + data[idx+1] + data[idx+2]) / 3;
      return brightness < 200; 
    };

    const castRay = (cx, cy, dx, dy, maxDist) => {
      let x = cx; let y = cy;
      for (let i = 0; i < maxDist; i++) {
        if (isWall(x, y)) return { dist: i, p: { x: Math.floor(x), y: Math.floor(y) } };
        x += dx; y += dy;
      }
      return { dist: maxDist, p: null };
    };

    const getInwardNormal = (x, y, originX, originY) => {
      let nx = 0, ny = 0; const radius = 4;
      for(let oy = -radius; oy <= radius; oy++) {
          for(let ox = -radius; ox <= radius; ox++) {
              if(ox === 0 && oy === 0) continue;
              if(isWall(x + ox, y + oy)) { nx -= ox; ny -= oy; }
          }
      }
      const mag = Math.hypot(nx, ny);
      if (mag > 0.1) {
          nx /= mag; ny /= mag;
          const dirX = originX - x, dirY = originY - y;
          if (nx * dirX + ny * dirY < 0) { nx = -nx; ny = -ny; }
          return {x: nx, y: ny};
      }
      const dirX = originX - x, dirY = originY - y;
      const dMag = Math.hypot(dirX, dirY);
      if(dMag === 0) return {x:0, y:0};
      return {x: dirX/dMag, y: dirY/dMag};
    };

    const getShortestChord = (cx, cy) => {
      let bestP1 = null, bestP2 = null;
      let shortestDist = Infinity;
      let bestAngle = 0;
      for (let angle = 0; angle < 180; angle += 10) {
        const rad = (angle * Math.PI) / 180;
        const dx = Math.cos(rad), dy = Math.sin(rad);

        const r1 = castRay(cx, cy, dx, dy, maxCut);
        const r2 = castRay(cx, cy, -dx, -dy, maxCut);

        if (r1.p && r2.p && r1.dist < maxCut && r2.dist < maxCut) {
          const totalDist = r1.dist + r2.dist;
          if (totalDist < shortestDist) {
            shortestDist = totalDist;
            bestP1 = r1.p; bestP2 = r2.p;
            bestAngle = angle;
          }
        }
      }
      return { dist: shortestDist, p1: bestP1, p2: bestP2, angle: bestAngle };
    };

    const candidates = [];
    const gridStep = Math.max(5, Math.floor(spacing / 3));

    for (let y = gridStep; y < h; y += gridStep) {
      for (let x = gridStep; x < w; x += gridStep) {
        if (!isWall(x, y)) {
          const chord = getShortestChord(x, y);
          if (chord.dist >= minCut && chord.dist <= maxCut) {
            const mx = (chord.p1.x + chord.p2.x) / 2;
            const my = (chord.p1.y + chord.p2.y) / 2;
            const perpRad = ((chord.angle + 90) % 360) * Math.PI / 180;
            const px = Math.cos(perpRad);
            const py = Math.sin(perpRad);
            
            const space1 = castRay(mx, my, px, py, maxCut).dist;
            const space2 = castRay(mx, my, -px, -py, maxCut).dist;
            const minDepth = Math.max(minCut * 1.5, chord.dist * 0.4);

            if (space1 > minDepth && space2 > minDepth) {
              candidates.push({ x, y, chord });
            }
          }
        }
      }
    }

    candidates.sort((a, b) => a.chord.dist - b.chord.dist);

    let drawnCount = 0;
    const drawnMidpoints = [];
    const newPaths = []; 

    const burnLineToMask = (x0, y0, x1, y1) => {
      const dist = Math.hypot(x1 - x0, y1 - y0);
      const steps = Math.max(1, Math.ceil(dist * 2)); 
      const dx = (x1 - x0) / steps; const dy = (y1 - y0) / steps;
      let x = x0, y = y0;
      const radius = Math.max(1, Math.floor(thick / 2));

      for (let i = 0; i <= steps; i++) {
        for (let oy = -radius; oy <= radius; oy++) {
          for (let ox = -radius; ox <= radius; ox++) {
            const px = Math.round(x) + ox; const py = Math.round(y) + oy;
            if (px >= 0 && px < w && py >= 0 && py < h) {
              const idx = (py * w + px) * 4;
              if (data[idx+3] >= 128) {
                data[idx] = 0; data[idx+1] = 0; data[idx+2] = 0; 
              }
            }
          }
        }
        x += dx; y += dy;
      }
    };

    for (const cand of candidates) {
      if (isWall(cand.x, cand.y)) continue;

      let tooClose = false;
      for (const mid of drawnMidpoints) {
        if (Math.hypot(cand.x - mid.x, cand.y - mid.y) < spacing) {
          tooClose = true; break;
        }
      }
      if (tooClose) continue;

      const liveChord = getShortestChord(cand.x, cand.y);
      
      if (liveChord.dist >= minCut && liveChord.dist <= maxCut) {
        const mx = (liveChord.p1.x + liveChord.p2.x) / 2;
        const my = (liveChord.p1.y + liveChord.p2.y) / 2;
        const perpRad = ((liveChord.angle + 90) % 360) * Math.PI / 180;
        const px = Math.cos(perpRad);
        const py = Math.sin(perpRad);
        
        const space1 = castRay(mx, my, px, py, maxCut).dist;
        const space2 = castRay(mx, my, -px, -py, maxCut).dist;
        const minDepth = Math.max(minCut * 1.5, liveChord.dist * 0.4);

        if (space1 > minDepth && space2 > minDepth) {
          const chordDist = liveChord.dist;
          const n1 = getInwardNormal(liveChord.p1.x, liveChord.p1.y, cand.x, cand.y);
          const n2 = getInwardNormal(liveChord.p2.x, liveChord.p2.y, cand.x, cand.y);

          const cp1x = liveChord.p1.x + n1.x * (chordDist * smoothness);
          const cp1y = liveChord.p1.y + n1.y * (chordDist * smoothness);
          const cp2x = liveChord.p2.x + n2.x * (chordDist * smoothness);
          const cp2y = liveChord.p2.y + n2.y * (chordDist * smoothness);

          newPaths.push({
            id: Date.now() + Math.random(),
            type: 'auto',
            isEraser: false,
            color: color,
            thickness: thick,
            p1: { x: liveChord.p1.x, y: liveChord.p1.y },
            p2: { x: liveChord.p2.x, y: liveChord.p2.y },
            cp1: smoothness > 0.02 ? { x: cp1x, y: cp1y } : null,
            cp2: smoothness > 0.02 ? { x: cp2x, y: cp2y } : null,
          });

          if (smoothness > 0.02) {
            let prevX = liveChord.p1.x, prevY = liveChord.p1.y;
            const steps = Math.max(8, Math.floor(chordDist)); 
            for (let i = 1; i <= steps; i++) {
              const t = i / steps, mt = 1 - t;
              const cx = mt*mt*mt*liveChord.p1.x + 3*mt*mt*t*cp1x + 3*mt*t*t*cp2x + t*t*t*liveChord.p2.x;
              const cy = mt*mt*mt*liveChord.p1.y + 3*mt*mt*t*cp1y + 3*mt*t*t*cp2y + t*t*t*liveChord.p2.y;
              burnLineToMask(prevX, prevY, cx, cy);
              prevX = cx; prevY = cy;
            }
          } else {
            burnLineToMask(liveChord.p1.x, liveChord.p1.y, liveChord.p2.x, liveChord.p2.y);
          }
          drawnMidpoints.push({ x: mx, y: my });
          drawnCount++;
        }
      }
    }

    renderAllPaths(maskCtx, w, h, newPaths);
    
    // Execute Final Pipeline Render logic through our central helper
    imgObj.paths = newPaths; 
    const result = await regenerateBlobsForImage(imgObj, newPaths);
    
    return {
        ...result,
        paths: newPaths,
        history: [newPaths],
        historyIndex: 0,
        autoStats: { thick, lines: drawnCount, time: Math.round(performance.now() - startTime) }
    };
  };

  // Automated Batch Logic with Support for Uniform vs Adaptive Modes
  const autoProcessBatch = async (batchImages, doCalibration) => {
    setIsProcessing(true);
    try {
        if (batchMode === 'uniform') {
            let baseParams = { thick: Number(lineThickness) || 1, minCut: Number(minCutLength) || 50, maxCut: Number(maxCutLength) || 200, spacing: Number(minSpacing) || 100, smoothness: Number(curveSmoothness) || 0, color: lineColor || '#000000' };
            
            if (doCalibration && batchImages.length > 0) {
                baseParams = await calibrateImage(batchImages[0]);
                setLineThickness(baseParams.thick);
                setMinCutLength(baseParams.minCut);
                setMaxCutLength(baseParams.maxCut);
                setMinSpacing(baseParams.spacing);
                
                setDefaultSettings(prev => ({
                  ...prev,
                  lineThickness: baseParams.thick, minCutLength: baseParams.minCut, maxCutLength: baseParams.maxCut, minSpacing: baseParams.spacing
                }));
            }

            for (let imgObj of batchImages) {
                imgObj.params = { ...baseParams };
                const result = await runAwareSegmenterCore(imgObj, imgObj.params);
                setImages(prev => prev.map(img => img.id === imgObj.id ? { ...img, ...result, params: imgObj.params } : img));
            }
        } else {
            // Adaptive Mode: Each image gets custom parameters
            for (let imgObj of batchImages) {
                let imgParams;
                if (doCalibration) {
                    imgParams = await calibrateImage(imgObj);
                } else {
                    imgParams = { thick: Number(lineThickness) || 1, minCut: Number(minCutLength) || 50, maxCut: Number(maxCutLength) || 200, spacing: Number(minSpacing) || 100, smoothness: Number(curveSmoothness) || 0, color: lineColor || '#000000' };
                }
                imgObj.params = imgParams;
                const result = await runAwareSegmenterCore(imgObj, imgParams);
                setImages(prev => prev.map(img => img.id === imgObj.id ? { ...img, ...result, params: imgParams } : img));
            }
            // Update global sliders to reflect the first image that is about to become actively viewed
            if (doCalibration && batchImages.length > 0) {
                const firstParams = batchImages[0].params;
                setLineThickness(firstParams.thick);
                setMinCutLength(firstParams.minCut);
                setMaxCutLength(firstParams.maxCut);
                setMinSpacing(firstParams.spacing);
            }
        }
    } catch(e) {
        console.error("AutoBatch Processing Failed:", e);
    }
    setIsProcessing(false);
  };

  const handleResetSettings = () => {
    setMinSpacing(defaultSettings.minSpacing);
    setMaxCutLength(defaultSettings.maxCutLength);
    setMinCutLength(defaultSettings.minCutLength);
    setLineThickness(defaultSettings.lineThickness);
    setCurveSmoothness(defaultSettings.curveSmoothness);
    handleLineColorChange(defaultSettings.lineColor);
    setViewerBg(defaultSettings.viewerBg);
    setHideLines(defaultSettings.hideLines);
    hideLinesRef.current = defaultSettings.hideLines;
    setLineBlendMode(defaultSettings.lineBlendMode);
    lineBlendModeRef.current = defaultSettings.lineBlendMode;
    
    setSvgCanvasSize(defaultSettings.svgCanvasSize);
    setSvgCustomW(defaultSettings.svgCustomW);
    setSvgCustomH(defaultSettings.svgCustomH);
    setSvgScaleMode(defaultSettings.svgScaleMode);
    setSvgStrokeWidth(defaultSettings.svgStrokeWidth);
  };

  const handleManualRun = async () => {
    if (!activeImg) return;
    setIsProcessing(true);
    setViewerMode('preview');
    try {
        const params = { 
            thick: Number(lineThickness) || 1, 
            minCut: Number(minCutLength) || 50, 
            maxCut: Number(maxCutLength) || 200, 
            spacing: Number(minSpacing) || 100, 
            smoothness: Number(curveSmoothness) || 0, 
            color: lineColor || '#000000' 
        };
        activeImg.params = params; // Bind manual run params to current image
        const result = await runAwareSegmenterCore(activeImg, params);
        updateActiveImage({ ...result, params });
    } catch(e) { console.error(e); }
    setIsProcessing(false);
  };

  const handleBatchRun = async () => {
    if (images.length === 0) return;
    setIsProcessing(true);
    setViewerMode('preview');
    try {
        const idsToProcess = imagesRef.current.map(img => img.id);
        for (let id of idsToProcess) {
            const imgObj = imagesRef.current.find(i => i.id === id);
            if (!imgObj) continue;
            
            let runParams;
            if (batchMode === 'uniform') {
                // Force current UI slider values onto all images
                runParams = { thick: Number(lineThickness) || 1, minCut: Number(minCutLength) || 50, maxCut: Number(maxCutLength) || 200, spacing: Number(minSpacing) || 100, smoothness: Number(curveSmoothness) || 0, color: lineColor || '#000000' };
            } else {
                // Adaptive: Use the specific parameters saved in each image (which could have been individually modified)
                runParams = imgObj.params || { thick: Number(lineThickness) || 1, minCut: Number(minCutLength) || 50, maxCut: Number(maxCutLength) || 200, spacing: Number(minSpacing) || 100, smoothness: Number(curveSmoothness) || 0, color: lineColor || '#000000' };
            }

            const result = await runAwareSegmenterCore(imgObj, runParams);
            setImages(prev => prev.map(img => img.id === id ? { ...img, ...result, params: runParams } : img));
        }
    } catch(e) { console.error(e); }
    setIsProcessing(false);
  };

  const handleDownload = (urlToDownload, typeLabel, originalName) => {
    if (!urlToDownload) return;
    const a = document.createElement('a');
    a.href = urlToDownload;
    a.download = `segmented-${typeLabel}-${originalName || Date.now()}.png`;
    document.body.appendChild(a);
    a.click();
    document.body.removeChild(a);
  };

  const loadJSZip = async () => {
    if (window.JSZip) return window.JSZip;
    return new Promise((resolve, reject) => {
      const script = document.createElement('script');
      script.src = 'https://cdnjs.cloudflare.com/ajax/libs/jszip/3.10.1/jszip.min.js';
      script.onload = () => resolve(window.JSZip);
      script.onerror = () => {
        console.error("Failed to load JSZip");
        reject(new Error("Failed to load JSZip"));
      };
      document.head.appendChild(script);
    });
  };

  const handleDownloadZip = async (type) => {
    setIsProcessing(true);
    setDropdownOpen(null);
    try {
      const JSZip = await loadJSZip();
      const zip = new JSZip();
      
      const targetImages = imagesRef.current.filter(img => type === 'mask' ? img.resultMaskURL : type === 'letters' ? img.resultLettersURL : img.resultImage);
      
      for (let i = 0; i < targetImages.length; i++) {
        const img = targetImages[i];
        const url = type === 'mask' ? img.resultMaskURL : type === 'letters' ? img.resultLettersURL : img.resultImage;
        const blob = await fetch(url).then(r => r.blob());
        const baseName = img.name ? img.name.replace(/\.[^/.]+$/, "") : `image-${i+1}`;
        zip.file(`segmented-${type}-${baseName}.png`, blob);
      }
      
      const zipBlob = await zip.generateAsync({ type: 'blob' });
      const zipUrl = URL.createObjectURL(zipBlob);
      
      const a = document.createElement('a');
      a.href = zipUrl;
      a.download = `segmented-${type}s-batch-${Date.now()}.zip`;
      document.body.appendChild(a);
      a.click();
      document.body.removeChild(a);
      URL.revokeObjectURL(zipUrl);
    } catch (err) {
      console.error("ZIP Generation Failed:", err);
    } finally {
      setIsProcessing(false);
    }
  };

  // --- 🖌 MANUAL EDITING LOGIC ---
  const startEditing = () => {
    setViewerMode('edit');
    setPenNodes([]);
    setPreviewPos(null);
    isDraggingHandle.current = false;
  };

  const saveEdits = async () => {
    setIsProcessing(true);
    try {
      if (penNodes.length > 0) commitPenPath(); 
      const currentPaths = historyRef.current[historyIndexRef.current];
      const newBlobs = await regenerateBlobsForImage(activeImg, currentPaths);
      
      updateActiveImage({ ...newBlobs, paths: currentPaths });
      setViewerMode('preview');
    } catch (e) {
      console.error("Failed saving edits", e);
    } finally {
      setIsProcessing(false);
    }
  };

  const cancelEdits = () => {
    setViewerMode('preview');
    setPenNodes([]);
    isDraggingHandle.current = false;
  };

  const getCanvasCoordinates = (e) => {
    if (!overlayCanvasRef.current) return { x: 0, y: 0 };
    const canvas = overlayCanvasRef.current;
    const rect = canvas.getBoundingClientRect();
    const clientX = e.touches ? e.touches[0].clientX : e.clientX;
    const clientY = e.touches ? e.touches[0].clientY : e.clientY;
    return { 
       x: (clientX - rect.left) / zoom, 
       y: (clientY - rect.top) / zoom 
    };
  };

  const commitPenPath = () => {
    if (penNodes.length < 2) {
      setPenNodes([]);
      setPreviewPos(null);
      isDraggingHandle.current = false;
      return;
    }
    
    const newPath = {
      id: crypto.randomUUID(),
      type: 'pen',
      isEraser: false,
      color: lineColor,
      thickness: Number(lineThickness) || 1,
      nodes: [...penNodes]
    };

    updatePathsAndHistory([...activeImg.paths, newPath]);
    setPenNodes([]);
    setPreviewPos(null);
    isDraggingHandle.current = false;
  };

  // Live render overlay
  useEffect(() => {
    if (viewerMode !== 'edit' || !overlayCanvasRef.current) return;
    const ctx = overlayCanvasRef.current.getContext('2d');
    const w = overlayCanvasRef.current.width;
    const h = overlayCanvasRef.current.height;
    ctx.clearRect(0, 0, w, h);

    const thick = Number(lineThickness) || 1;
    const eSize = Number(eraserSize) || 9;

    if ((editTool === 'brush' || (editTool === 'erase' && eraserMode === 'brush')) && previewPos && !isSpacePressed) {
      const radius = (editTool === 'erase' ? eSize : thick) / 2;
      
      ctx.beginPath();
      ctx.arc(previewPos.x, previewPos.y, radius + (1 / zoom), 0, Math.PI * 2);
      ctx.lineWidth = 1.5 / zoom; 
      ctx.strokeStyle = 'rgba(255, 255, 255, 0.8)';
      ctx.stroke();

      ctx.beginPath();
      ctx.arc(previewPos.x, previewPos.y, radius, 0, Math.PI * 2);
      ctx.lineWidth = 1.5 / zoom;
      ctx.strokeStyle = editTool === 'erase' ? '#000000' : lineColor;
      ctx.stroke();
    }

    if (editTool === 'pen' && penNodes.length > 0) {
      ctx.beginPath();
      ctx.strokeStyle = lineColor;
      ctx.lineWidth = thick;
      ctx.lineCap = 'round';
      ctx.lineJoin = 'round';
      
      ctx.moveTo(penNodes[0].x, penNodes[0].y);
      for (let i = 1; i < penNodes.length; i++) {
        const prev = penNodes[i - 1];
        const curr = penNodes[i];
        ctx.bezierCurveTo(prev.handleOut.x, prev.handleOut.y, curr.handleIn.x, curr.handleIn.y, curr.x, curr.y);
      }

      if (previewPos && !isDraggingHandle.current) {
        const prev = penNodes[penNodes.length - 1];
        ctx.bezierCurveTo(prev.handleOut.x, prev.handleOut.y, previewPos.x, previewPos.y, previewPos.x, previewPos.y);
      }
      ctx.stroke();

      const handleScale = 1 / zoom; 
      
      penNodes.forEach((p, idx) => {
        const hasPulledHandle = p.handleOut.x !== p.x || p.handleOut.y !== p.y;
        if (hasPulledHandle) {
          ctx.beginPath();
          ctx.moveTo(p.handleIn.x, p.handleIn.y);
          ctx.lineTo(p.handleOut.x, p.handleOut.y);
          ctx.strokeStyle = '#a5b4fc';
          ctx.lineWidth = 1 * handleScale;
          ctx.stroke();

          ctx.beginPath();
          ctx.arc(p.handleIn.x, p.handleIn.y, 3 * handleScale, 0, Math.PI * 2);
          ctx.arc(p.handleOut.x, p.handleOut.y, 3 * handleScale, 0, Math.PI * 2);
          ctx.fillStyle = '#ffffff';
          ctx.strokeStyle = '#6366f1';
          ctx.fill();
          ctx.stroke();
        }

        ctx.beginPath();
        ctx.rect(p.x - (3 * handleScale), p.y - (3 * handleScale), 6 * handleScale, 6 * handleScale);
        ctx.fillStyle = idx === penNodes.length - 1 ? '#6366f1' : '#ffffff';
        ctx.strokeStyle = '#6366f1';
        ctx.lineWidth = 1.5 * handleScale;
        ctx.fill();
        ctx.stroke();
      });
    }
  }, [penNodes, previewPos, editTool, viewerMode, lineColor, lineThickness, zoom, eraserSize, eraserMode, isSpacePressed]);

  const handleMagicEraseVector = (x, y) => {
    if (!activeImg) return;
    const ctx = editCanvasRef.current.getContext('2d');
    
    let hitIndex = -1;
    const currentPaths = activeImg.paths || [];

    for (let i = currentPaths.length - 1; i >= 0; i--) {
      const p = currentPaths[i];
      if (p.isEraser) continue; 

      const path2d = buildPath2D(p);
      ctx.lineWidth = p.thickness + 24; 
      if (ctx.isPointInStroke(path2d, x, y)) {
          hitIndex = i;
          break;
      }
    }

    if (hitIndex !== -1) {
      const newPaths = [...currentPaths];
      newPaths.splice(hitIndex, 1);
      updatePathsAndHistory(newPaths);
    }
  };

  const handleEditMouseDown = (e) => {
    if (isSpacePressed || e.button === 1 || e.button === 2) return; 
    e.stopPropagation();

    const { x, y } = getCanvasCoordinates(e);
    const thick = Number(lineThickness) || 1;
    const eSize = Number(eraserSize) || 9;
    
    if (editTool === 'pen') {
      isDraggingHandle.current = true;
      setPenNodes(prev => [...prev, { x, y, handleIn: { x, y }, handleOut: { x, y } }]);
    } else if (editTool === 'erase' && eraserMode === 'magic') {
      isMagicErasing.current = true;
      handleMagicEraseVector(x, y); 
    } else {
      setIsDrawing(true);
      drawPoints.current = [{ x, y }];
      lastDrawnPos.current = { x, y };

      const ctx = editCanvasRef.current.getContext('2d');
      ctx.lineCap = 'round';
      ctx.lineJoin = 'round';
      
      if (editTool === 'erase') {
        ctx.globalCompositeOperation = 'destination-out';
        ctx.strokeStyle = 'rgba(0,0,0,1)'; 
        ctx.fillStyle = 'rgba(0,0,0,1)';
        ctx.lineWidth = eSize;
      } else {
        ctx.globalCompositeOperation = 'source-over';
        ctx.strokeStyle = lineColor;
        ctx.fillStyle = lineColor;
        ctx.lineWidth = thick;
      }
      
      const radius = (editTool === 'erase' ? eSize : thick) / 2;
      ctx.beginPath();
      ctx.arc(x, y, radius, 0, Math.PI * 2);
      ctx.fill();
    }
  };

  const handleEditMouseMove = (e) => {
    if (isSpacePressed) {
      setPreviewPos(null); 
      return;
    }
    const { x, y } = getCanvasCoordinates(e);

    if (editTool === 'erase' && eraserMode === 'magic') {
      if (isMagicErasing.current) {
        handleMagicEraseVector(x, y);
      }
    } else if (editTool === 'pen') {
      if (isDraggingHandle.current) {
        setPenNodes(prev => {
          const newNodes = [...prev];
          const lastNode = newNodes[newNodes.length - 1];
          lastNode.handleOut = { x, y };
          lastNode.handleIn = { x: lastNode.x * 2 - x, y: lastNode.y * 2 - y };
          return newNodes;
        });
      } else {
        setPreviewPos({ x, y });
      }
    } else if (editTool === 'brush' || (editTool === 'erase' && eraserMode === 'brush')) {
      setPreviewPos({ x, y }); 
      
      if (isDrawing) {
        const lastPt = drawPoints.current[drawPoints.current.length - 1];
        const dist = Math.hypot(x - lastPt.x, y - lastPt.y);
        const skipThreshold = (freehandSmoothness / 100) * 20;

        if (dist > skipThreshold || freehandSmoothness === 0) {
           drawPoints.current.push({ x, y });
           const pts = drawPoints.current;
           
           if (pts.length >= 3 && lastDrawnPos.current) {
             const ctx = editCanvasRef.current.getContext('2d');
             const p2 = pts[pts.length - 2];
             const p3 = pts[pts.length - 1];
             const mid = { x: (p2.x + p3.x) / 2, y: (p2.y + p3.y) / 2 };

             ctx.beginPath();
             ctx.moveTo(lastDrawnPos.current.x, lastDrawnPos.current.y);
             ctx.quadraticCurveTo(p2.x, p2.y, mid.x, mid.y);
             ctx.stroke();

             lastDrawnPos.current = mid;
           }
        }
      }
    }
  };

  const handleEditMouseUp = (e) => {
    isMagicErasing.current = false;
    
    if (editTool === 'pen') {
      if (isDraggingHandle.current) {
        isDraggingHandle.current = false;
        const { x, y } = getCanvasCoordinates(e);
        setPreviewPos({ x, y }); 
      }
      return;
    }

    if (!isDrawing) return;
    setIsDrawing(false);
    
    if (previewPos) {
       drawPoints.current.push(previewPos);
    }

    const pts = [...drawPoints.current];
    if (pts.length > 0 && activeImg) {
       const newPath = {
          id: crypto.randomUUID(),
          type: 'brush',
          isEraser: editTool === 'erase',
          color: editTool === 'erase' ? null : lineColor,
          thickness: editTool === 'erase' ? (Number(eraserSize) || 9) : (Number(lineThickness) || 1),
          smoothness: freehandSmoothness,
          points: pts
       };
       updatePathsAndHistory([...activeImg.paths, newPath]);

                <button onClick={handleZoomOut} disabled={zoom <= minZoom} className="p-1.5 text-neutral-600 hover:bg-white rounded disabled:opacity-30"><ZoomOut className="w-4 h-4"/></button>
                <span className="text-xs font-semibold text-neutral-600 w-12 text-center">{Math.round(zoom * 100)}%</span>
                <button onClick={handleZoomIn} disabled={zoom >= 5} className="p-1.5 text-neutral-600 hover:bg-white rounded disabled:opacity-30"><ZoomIn className="w-4 h-4"/></button>
                <div className="w-px h-4 bg-neutral-300 mx-1"></div>
                <button onClick={handleFitScreen} className="p-1.5 text-neutral-600 hover:bg-white rounded" title="Fit to Screen"><Maximize className="w-4 h-4"/></button>
              </div>

              {activeImg && activeImg.resultImage && (
                <div className="flex bg-neutral-200/60 p-0.5 rounded-md border border-neutral-200 shadow-inner">
                  <button
                    onClick={() => setShowProcessed(false)}
                    className={`px-3 py-1.5 text-xs font-semibold rounded-sm transition-all ${!showProcessed ? 'bg-white shadow-sm text-indigo-700' : 'text-neutral-500 hover:text-neutral-700 hover:bg-neutral-100/50'}`}
                  >
                    Original
                  </button>
                  <button
                    onClick={() => setShowProcessed(true)}
                    className={`px-3 py-1.5 text-xs font-semibold rounded-sm transition-all ${showProcessed ? 'bg-white shadow-sm text-indigo-700' : 'text-neutral-500 hover:text-neutral-700 hover:bg-neutral-100/50'}`}
                  >
                    Processed
                  </button>
                </div>
              )}
            </div>
            
            {/* Right side: Tools & Export */}
            <div className="flex gap-2 relative z-50">
              {dropdownOpen && (
                <div className="fixed inset-0 z-40" onClick={() => setDropdownOpen(null)} />
              )}

              {activeImg && activeImg.resultImage && activeImg.resultMaskURL && (
                <>
                  <button 
                    onClick={startEditing}
                    className="relative z-50 text-xs flex items-center gap-1.5 px-3 py-1.5 bg-purple-50 border border-purple-200 rounded-md hover:bg-purple-100 text-purple-700 font-medium transition-colors shadow-sm"
                    title="Draw or erase lines manually"
                  >
                    <PenTool className="w-3.5 h-3.5" /> Refine Current
                  </button>
                  <div className="w-px bg-neutral-300 mx-1 z-50"></div>
                  
                  {/* Save Mask Dropdown */}
                  <div className="relative z-50">
                    <button 
                      onClick={() => setDropdownOpen(dropdownOpen === 'mask' ? null : 'mask')}
                      className="text-xs flex items-center gap-1.5 px-3 py-1.5 bg-white border border-neutral-300 rounded-md hover:bg-neutral-50 text-neutral-700 font-medium transition-colors shadow-sm"
                    >
                      <Layers className="w-3.5 h-3.5" /> Save Mask <ChevronDown className="w-3 h-3 ml-0.5" />
                    </button>
                    {dropdownOpen === 'mask' && (
                      <div className="absolute top-full right-0 mt-1 bg-white border border-gray-200 shadow-lg rounded-md w-48 py-1">
                        <button 
                          onClick={() => handleDownloadZip('mask')} 
                          className="w-full text-left px-4 py-2 text-sm text-gray-700 hover:bg-gray-100 flex items-center gap-2"
                        >
                          <Images className="w-4 h-4"/> All Masks (ZIP)
                        </button>
                        <button 
                          onClick={() => { handleDownload(activeImg.resultMaskURL, 'mask-only', activeImg.name); setDropdownOpen(null); }} 
                          className="w-full text-left px-4 py-2 text-sm text-gray-700 hover:bg-gray-100 flex items-center gap-2"
                        >
                          <Layers className="w-4 h-4"/> Current Mask Only
                        </button>
                      </div>
                    )}
                  </div>

                  {/* Save Letters Dropdown (Only show if enabled) */}
                  {letterSettings.enabled && activeImg.resultLettersURL && (
                    <div className="relative z-50">
                      <button 
                        onClick={() => setDropdownOpen(dropdownOpen === 'letters' ? null : 'letters')}
                        className="text-xs flex items-center gap-1.5 px-3 py-1.5 bg-white border border-neutral-300 rounded-md hover:bg-neutral-50 text-neutral-700 font-medium transition-colors shadow-sm"
                      >
                        <Type className="w-3.5 h-3.5" /> Save Letters <ChevronDown className="w-3 h-3 ml-0.5" />
                      </button>
                      {dropdownOpen === 'letters' && (
                        <div className="absolute top-full right-0 mt-1 bg-white border border-gray-200 shadow-lg rounded-md w-48 py-1">
                          <button 
                            onClick={() => handleDownloadZip('letters')} 
                            className="w-full text-left px-4 py-2 text-sm text-gray-700 hover:bg-gray-100 flex items-center gap-2"
                          >
                            <Images className="w-4 h-4"/> All Letters (ZIP)
                          </button>
                          <button 
                            onClick={() => { handleDownload(activeImg.resultLettersURL, 'letters-only', activeImg.name); setDropdownOpen(null); }} 
                            className="w-full text-left px-4 py-2 text-sm text-gray-700 hover:bg-gray-100 flex items-center gap-2"
                          >
                            <Type className="w-4 h-4"/> Current Letters Only
                          </button>
                        </div>
                      )}
                    </div>
                  )}

                  {/* Save Art Dropdown */}
                  <div className="relative z-50">
                    <button 
                      onClick={() => setDropdownOpen(dropdownOpen === 'art' ? null : 'art')}
                      className="text-xs flex items-center gap-1.5 px-3 py-1.5 bg-white border border-neutral-300 rounded-md hover:bg-neutral-50 text-neutral-700 font-medium transition-colors shadow-sm"
                    >
                      <Download className="w-3.5 h-3.5" /> Save Art <ChevronDown className="w-3 h-3 ml-0.5" />
                    </button>
                    {dropdownOpen === 'art' && (
                      <div className="absolute top-full right-0 mt-1 bg-white border border-gray-200 shadow-lg rounded-md w-48 py-1">
                        <button 
                          onClick={() => handleDownloadZip('art')} 
                          className="w-full text-left px-4 py-2 text-sm text-gray-700 hover:bg-gray-100 flex items-center gap-2"
                        >
                          <Images className="w-4 h-4"/> All Art Pages (ZIP)
                        </button>
                        <button 
                          onClick={() => { handleDownload(activeImg.resultImage, 'full-art', activeImg.name); setDropdownOpen(null); }} 
                          className="w-full text-left px-4 py-2 text-sm text-gray-700 hover:bg-gray-100 flex items-center gap-2"
                        >
                          <Download className="w-4 h-4"/> Current Art Only
                        </button>
                      </div>
                    )}
                  </div>
                </>
              )}
            </div>
          </div>
        ) : (
          <div className="h-14 bg-purple-50 border-b border-purple-200 flex justify-between items-center px-4 shadow-sm z-10 flex-shrink-0">
            <div className="flex items-center gap-4">
              <span className="text-sm font-bold text-purple-700 flex items-center gap-2">
                <PenTool className="w-4 h-4" />
                Refine Mode
              </span>

              <div className="flex items-center gap-1 bg-white/50 rounded-md border border-purple-200 p-0.5">
                <button onClick={handleZoomOut} disabled={zoom <= minZoom} className="p-1.5 text-purple-700 hover:bg-white rounded disabled:opacity-30"><ZoomOut className="w-4 h-4"/></button>
                <span className="text-xs font-semibold text-purple-800 w-12 text-center">{Math.round(zoom * 100)}%</span>
                <button onClick={handleZoomIn} disabled={zoom >= 5} className="p-1.5 text-purple-700 hover:bg-white rounded disabled:opacity-30"><ZoomIn className="w-4 h-4"/></button>
                <div className="w-px h-4 bg-purple-200 mx-1"></div>
                <button onClick={handleFitScreen} className="p-1.5 text-purple-700 hover:bg-white rounded" title="Fit to Screen"><Maximize className="w-4 h-4"/></button>
              </div>
            </div>
            
            <div className="flex items-center gap-2">
              
              {/* Undo/Redo Engine Controls */}
              <div className="flex items-center gap-1 bg-white/50 rounded-md border border-purple-200 p-0.5 mr-1">
                <button 
                  onClick={handleUndo} 
                  disabled={!canUndo} 
                  className="p-1.5 text-purple-700 hover:bg-white rounded disabled:opacity-30"
                  title="Undo (Ctrl+Z)"
                >
                  <Undo2 className="w-4 h-4"/>
                </button>
                <button 
                  onClick={handleRedo} 
                  disabled={!canRedo} 
                  className="p-1.5 text-purple-700 hover:bg-white rounded disabled:opacity-30"
                  title="Redo (Ctrl+Shift+Z)"
                >
                  <Redo2 className="w-4 h-4"/>
                </button>
              </div>

              <div className="flex bg-white border border-neutral-300 rounded-md shadow-sm overflow-hidden">
                <button 
                  onClick={() => { setEditTool('pen'); setPenNodes([]); setPreviewPos(null); }} 
                  className={`p-1.5 transition-colors ${editTool === 'pen' ? 'bg-purple-100 text-purple-700' : 'text-neutral-500 hover:bg-neutral-50'}`}
                  title="Vector Pen (Click points. Drag to curve. Right-click to finish)"
                >
                  <PenTool className="w-4 h-4" />
                </button>
                <button 
                  onClick={() => { setEditTool('brush'); setPenNodes([]); setPreviewPos(null); }} 
                  className={`p-1.5 border-l border-neutral-200 transition-colors ${editTool === 'brush' ? 'bg-purple-100 text-purple-700' : 'text-neutral-500 hover:bg-neutral-50'}`}
                  title="Freehand Brush (Drag to draw)"
                >
                  <Pencil className="w-4 h-4" />
                </button>
                <button 
                  onClick={() => { setEditTool('erase'); setPenNodes([]); setPreviewPos(null); }} 
                  className={`p-1.5 border-l border-neutral-200 transition-colors ${editTool === 'erase' ? 'bg-purple-100 text-purple-700' : 'text-neutral-500 hover:bg-neutral-50'}`}
                  title="Eraser (Drag to delete)"
                >
                  <Eraser className="w-4 h-4" />
                </button>
              </div>

              {editTool === 'pen' && penNodes.length > 0 && (
                <div className="flex bg-white border border-neutral-300 rounded-md shadow-sm overflow-hidden items-center ml-1">
                  <button 
                    id="finish-path-btn"
                    onClick={commitPenPath} 
                    className="text-xs px-3 py-1.5 bg-indigo-600 hover:bg-indigo-700 text-white font-medium transition-colors"
                    title="Press Enter or Escape to finish"
                  >
                    Finish Path (Enter)
                  </button>
                </div>
              )}

              {editTool === 'erase' && (
                <div className="flex bg-white border border-neutral-300 rounded-md shadow-sm overflow-hidden items-center ml-1">
                  <button 
                    onClick={() => setEraserMode('brush')} 
                    className={`text-xs font-semibold px-2 py-1.5 transition-colors ${eraserMode === 'brush' ? 'bg-indigo-100 text-indigo-700' : 'text-neutral-500 hover:bg-neutral-50'}`}
                    title="Erase manually by drawing over lines"
                  >
                    Brush
                  </button>
                  <button 
                    onClick={() => setEraserMode('magic')} 
                    className={`text-xs font-semibold px-2 py-1.5 border-l border-neutral-200 transition-colors ${eraserMode === 'magic' ? 'bg-indigo-100 text-indigo-700' : 'text-neutral-500 hover:bg-neutral-50'}`}
                    title="Click any line to instantly delete the entire vector segment"
                  >
                    Magic Erase
                  </button>
                </div>
              )}

              {/* Brush Smoothness Slider */}
              {(editTool === 'brush' || (editTool === 'erase' && eraserMode === 'brush')) && (
                <div className="flex items-center gap-2 bg-white px-2 py-1.5 rounded-md border border-neutral-200 shadow-sm ml-1 min-w-[120px]">
                  <span className="text-xs font-medium text-neutral-500">Smooth:</span>
                  <input 
                    type="range" min="0" max="100" step="1" value={freehandSmoothness} 
                    onChange={e => setFreehandSmoothness(Number(e.target.value))} 
                    className="w-16 accent-purple-600 flex-1"
                    title="Higher values stabilize jagged mouse movements"
                  />
                </div>
              )}
              
              <div className="flex items-center gap-2 bg-white px-2 py-1.5 rounded-md border border-neutral-200 shadow-sm ml-1 min-w-[100px]">
                <span className="text-xs font-medium text-neutral-500">Size:</span>
                {editTool === 'erase' && eraserMode === 'brush' ? (
                  <input 
                    type="range" min="1" max="50" step="0.5" value={eraserSize} 
                    onChange={e => setEraserSize(Number(e.target.value))} 
                    className="w-16 accent-purple-600 flex-1"
                  />
                ) : editTool === 'erase' && eraserMode === 'magic' ? (
                  <span className="text-xs font-bold text-neutral-400 flex-1 text-center">Auto</span>
                ) : (
                  <span className="text-xs font-bold text-neutral-700 flex-1 text-center">{lineThickness}px</span>
                )}
              </div>

              <div className="flex gap-2 ml-2 border-l border-purple-200 pl-3">
                <button 
                  onClick={cancelEdits} 
                  className="text-xs flex items-center gap-1 px-3 py-1.5 bg-white border border-neutral-300 rounded-md hover:bg-neutral-50 text-neutral-600 font-medium transition-colors"
                >
                  <X className="w-3.5 h-3.5" /> Cancel
                </button>
                <button 
                  id="apply-btn"
                  onClick={saveEdits} 
                  disabled={isProcessing}
                  className="text-xs flex items-center gap-1 px-3 py-1.5 bg-purple-600 border border-purple-700 rounded-md hover:bg-purple-700 text-white font-medium transition-colors shadow-sm disabled:opacity-50"
                  title="Save and Apply (Ctrl+Enter)"
                >
                  {isProcessing ? <Loader2 className="w-3.5 h-3.5 animate-spin" /> : <Check className="w-3.5 h-3.5" />}
                  Apply
                </button>
              </div>
            </div>
          </div>
        )}

        {/* Workspace Physical Artboard Area */}
        <div 
          id="preview-workspace" 
          className={`flex-1 overflow-auto relative outline-none ${isPanning ? 'cursor-grabbing' : isSpacePressed ? 'cursor-grab' : ''}`}
          ref={workspaceRef}
          onMouseDown={handleWorkspaceMouseDown}
        >
          {/* Inner container scales up with zoom */}
          <div 
            className="relative transition-opacity duration-200"
            style={{ 
               width: Math.max(workspaceSize.w, imageSize.w * zoom + 60), 
               height: Math.max(workspaceSize.h, imageSize.h * zoom + 60),
               opacity: isProcessing ? 0.6 : 1 
            }}
          >
            
            {/* Physical bounding box of the art */}
            <div 
              id="physical-page"
              className="absolute shadow-2xl origin-center"
              onContextMenu={viewerMode === 'edit' ? handleEditContextMenu : undefined}
              style={{
                left: '50%',
                top: '50%',
                transform: 'translate(-50%, -50%)',
                width: imageSize.w * zoom,
                height: imageSize.h * zoom,
                backgroundColor: viewerBg === 'transparent' ? '#ffffff' : viewerBg,
                backgroundImage: viewerBg === 'transparent' ? 'radial-gradient(#e5e7eb 1px, transparent 1px)' : 'none', 
                backgroundSize: `${20 * zoom}px ${20 * zoom}px`
              }}
            >
              
              {!activeImg && (
                <div className="absolute inset-0 flex flex-col items-center justify-center text-neutral-400">
                  <ImageIcon className="w-12 h-12 opacity-30 mb-2" />
                  <p className="text-xs font-semibold">Workspace (Upload to begin)</p>
                </div>
              )}

              {/* Preview Mode Render */}
              {activeImg && viewerMode === 'preview' && (
                <img 
                  src={showProcessed && activeImg.resultImage ? activeImg.resultImage : activeImg.dataUrl} 
                  alt="Artboard" 
                  className="w-full h-full object-fill pointer-events-none" 
                />
              )}

              {/* Edit Mode Render */}
              {viewerMode === 'edit' && activeImg && (
                <>
                  <img 
                    src={activeImg.dataUrl} 
                    alt="Base" 
                    className="w-full h-full object-fill pointer-events-none" 
                  />
                  
                  {/* Commits applied here */}
                  <canvas
                    ref={editCanvasRef}
                    width={imageSize.w}
                    height={imageSize.h}
                    className="absolute inset-0 object-fill pointer-events-none"
                    style={{ width: '100%', height: '100%', mixBlendMode: lineBlendMode === 'multiply' ? 'multiply' : 'normal' }}
                  />
                  
                  {/* Interactive overlay */}
                  <canvas
                    ref={overlayCanvasRef}
                    width={imageSize.w}
                    height={imageSize.h}
                    className={`absolute inset-0 object-fill touch-none ${
                      isSpacePressed ? '' :
                      (editTool === 'brush' || (editTool === 'erase' && eraserMode === 'brush')) ? 'cursor-none' : 
                      'cursor-crosshair'
                    }`}
                    style={{ width: '100%', height: '100%', mixBlendMode: lineBlendMode === 'multiply' ? 'multiply' : 'normal' }}
                    onMouseDown={handleEditMouseDown}
                    onMouseMove={handleEditMouseMove}
                    onMouseUp={handleEditMouseUp}
                    onMouseLeave={handleEditMouseLeave}
                    onTouchStart={handleEditMouseDown}
                    onTouchMove={handleEditMouseMove}
                    onTouchEnd={handleEditMouseUp}
                  />
                </>
              )}

            </div>
          </div>
        </div>

        {/* Filmstrip / Multi-Image Dock */}
        {images.length > 0 && (
          <div className="h-24 bg-white border-t border-gray-300 flex items-center px-4 gap-3 overflow-x-auto flex-shrink-0 z-20 shadow-[0_-4px_10px_rgba(0,0,0,0.02)] scrollbar-hide">
            {images.map((img, i) => (
               <div 
                 key={img.id} 
                 id={`thumb-${i}`}
                 className={`group relative bg-neutral-900 w-16 h-16 rounded-md border-2 overflow-hidden flex-shrink-0 cursor-pointer transition-all ${i === activeIndex ? 'border-indigo-600 shadow-md scale-105' : 'border-gray-200 hover:border-indigo-400'}`}
                 onClick={() => {
                    if (viewerMode === 'edit') setViewerMode('preview'); // safe exit from edit mode
                    setActiveIndex(i);
                 }}
                 title={img.name}
               >
                  <img src={img.dataUrl} className="w-full h-full object-contain" alt="thumbnail" />
                  
                  {img.resultImage && (
                     <div className="absolute bottom-1.5 left-1.5 bg-emerald-500 rounded-full w-2.5 h-2.5 border-2 border-white shadow-sm" title="Processed"></div>
                  )}
                  <button 
                    onClick={(e) => { e.stopPropagation(); deleteImage(i); }}
                    className="absolute top-1 right-1 bg-red-500 text-white p-0.5 rounded-full opacity-0 group-hover:opacity-100 transition-all hover:bg-red-600 hover:scale-110 shadow-sm flex items-center justify-center"
                    title="Remove Image"
                  >
                    <X className="w-3 h-3" />
                  </button>
               </div>
            ))}
            
            <label className="w-16 h-16 rounded-md border-2 border-dashed border-gray-300 flex items-center justify-center text-gray-400 hover:text-indigo-600 hover:border-indigo-400 hover:bg-indigo-50 cursor-pointer flex-shrink-0 transition-colors ml-1" title="Add more images">
               <input type="file" multiple accept="image/png, image/jpeg, image/webp, image/svg+xml" className="hidden" onChange={handleFileUpload} />
               <Plus className="w-6 h-6" />
            </label>
          </div>
        )}
      </div>
    </div>
  );
}
