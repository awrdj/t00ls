// Lightweight Binary Parser to extract true DPI from PNG/JPEG headers
const extractDPI = (dataUrl) => {
  try {
    const marker = dataUrl.split(',')[0];
    const b64Part = dataUrl.split(',')[1].substring(0, 4000); 
    const bstr = atob(b64Part);
    
    if (marker.includes('image/png')) {
      let offset = 8;
      while (offset < bstr.length - 12) {
        const length = (bstr.charCodeAt(offset) << 24) | (bstr.charCodeAt(offset+1) << 16) | (bstr.charCodeAt(offset+2) << 8) | bstr.charCodeAt(offset+3);
        const type = bstr.substring(offset+4, offset+8);
        if (type === 'pHYs') {
          const ppuX = (bstr.charCodeAt(offset+8) << 24) | (bstr.charCodeAt(offset+9) << 16) | (bstr.charCodeAt(offset+10) << 8) | bstr.charCodeAt(offset+11);
          const unit = bstr.charCodeAt(offset+16);
          if (unit === 1) return Math.round(ppuX * 0.0254); 
        }
        if (type === 'IDAT') break;
        offset += 12 + length;
      }
    } else if (marker.includes('image/jpeg')) {
      if (bstr.charCodeAt(0) === 0xFF && bstr.charCodeAt(1) === 0xD8) {
        let offset = 2;
        while (offset < bstr.length - 4) {
          if (bstr.charCodeAt(offset) !== 0xFF) break;
          const markerType = bstr.charCodeAt(offset+1);
          if (markerType === 0xE0) { 
            if (bstr.substring(offset+4, offset+8) === 'JFIF') {
              const units = bstr.charCodeAt(offset+11);
              const xDensity = (bstr.charCodeAt(offset+12) << 8) | bstr.charCodeAt(offset+13);
              if (units === 1) return xDensity; 
              if (units === 2) return Math.round(xDensity * 2.54); 
            }
          }
          const length = (bstr.charCodeAt(offset+2) << 8) | bstr.charCodeAt(offset+3);
          offset += 2 + length;
        }
      }
    }
  } catch (e) { 
    console.error('DPI Extraction Error', e); 
  }
  return 72; 
};

// --- CORE VECTOR PATH BUILDER ---
const buildPath2D = (path) => {
  const p = new Path2D();
  if (path.type === 'auto') {
    p.moveTo(path.p1.x, path.p1.y);
    if (path.cp1) p.bezierCurveTo(path.cp1.x, path.cp1.y, path.cp2.x, path.cp2.y, path.p2.x, path.p2.y);
    else p.lineTo(path.p2.x, path.p2.y);
  } else if (path.type === 'pen' && path.nodes && path.nodes.length > 0) {
    p.moveTo(path.nodes[0].x, path.nodes[0].y);
    for (let i = 1; i < path.nodes.length; i++) {
      p.bezierCurveTo(path.nodes[i-1].handleOut.x, path.nodes[i-1].handleOut.y, path.nodes[i].handleIn.x, path.nodes[i].handleIn.y, path.nodes[i].x, path.nodes[i].y);
    }
  } else if (path.type === 'brush' && path.points && path.points.length > 0) {
    p.moveTo(path.points[0].x, path.points[0].y);
    if (path.points.length > 2 && path.smoothness > 0) {
      for (let i = 2; i < path.points.length; i++) {
        const p2 = path.points[i - 1];
        const p3 = path.points[i];
        const mid = { x: (p2.x + p3.x) / 2, y: (p2.y + p3.y) / 2 };
        p.quadraticCurveTo(p2.x, p2.y, mid.x, mid.y);
      }
      p.lineTo(path.points[path.points.length - 1].x, path.points[path.points.length - 1].y);
    } else {
      for (let i = 1; i < path.points.length; i++) {
        p.lineTo(path.points[i].x, path.points[i].y);
      }
    }
  }
  return p;
};

// --- RENDER ENGINE ---
const OVERFLOW_MARGIN = 500; // Invisible margin for drawing outside image bounds (lowered for massive performance boost) was 2000

const renderAllPaths = (ctx, w, h, pathsToRender, offsetX = 0, offsetY = 0) => {
  ctx.clearRect(0, 0, w, h);
  ctx.save();
  if (offsetX || offsetY) ctx.translate(offsetX, offsetY); // Shift rendering to account for oversized canvas

  pathsToRender.forEach(p => {
    ctx.globalCompositeOperation = p.isEraser ? 'destination-out' : 'source-over';
    ctx.strokeStyle = p.color;
    ctx.fillStyle = p.color;
    ctx.lineWidth = p.thickness;
    ctx.lineCap = 'round';
    ctx.lineJoin = 'round';

    const p2d = buildPath2D(p);
    ctx.stroke(p2d);

    if (p.type === 'brush' && p.points.length === 1) {
       ctx.beginPath();
       ctx.arc(p.points[0].x, p.points[0].y, p.thickness / 2, 0, Math.PI * 2);
       ctx.fill();
    }
  });
  
  ctx.restore();
  ctx.globalCompositeOperation = 'source-over';
};

// Global cache for spiral offsets so we don't recalculate 40,000 items per image
let cachedSpiralOffsets = null;
const getSpiralOffsets = () => {
  if (cachedSpiralOffsets) return cachedSpiralOffsets;
  const step = 3; const maxNudge = 300; const offsets = [];
  for(let dy = -maxNudge; dy <= maxNudge; dy += step) {
      for(let dx = -maxNudge; dx <= maxNudge; dx += step) {
          offsets.push({ dx, dy, dist: dx*dx + dy*dy });
      }
  }
  cachedSpiralOffsets = offsets.sort((a,b) => a.dist - b.dist);
  return cachedSpiralOffsets;
};

// --- LETTER PLACER ALGORITHM (Python Logic -> JS Native) ---
const placeLetters = async (maskCanvas, imgObj, settings) => {
  // Force font load explicitly to guarantee rendering before measuring
  try {
    const primaryFont = settings.font.split(',')[0].replace(/['"]/g, '').trim();
    await document.fonts.load(`bold 20px "${primaryFont}"`);
  } catch (err) { console.warn('Font load trigger failed'); }
  await document.fonts.ready;

  const { w, h } = imgObj;
  const lettersCanvas = document.createElement('canvas');
  lettersCanvas.width = w; lettersCanvas.height = h;
  const ctxL = lettersCanvas.getContext('2d');

  const combinedCanvas = document.createElement('canvas');
  combinedCanvas.width = w; combinedCanvas.height = h;
  const ctxC = combinedCanvas.getContext('2d', { willReadFrequently: true });
  
  const baseImg = new Image();
  baseImg.src = imgObj.dataUrl;
  await new Promise(r => { baseImg.onload = r; });
  
  ctxC.drawImage(baseImg, 0, 0);
  const baseData = ctxC.getImageData(0, 0, w, h).data;
  
  // Read exact mask to prevent colored lines (like red) from breaking the flood fill
  const maskData = maskCanvas.getContext('2d').getImageData(0, 0, w, h).data;
  
  const labels = new Int32Array(w * h);
  const regions = [];
  // Pre-allocate a single, massive memory block to completely bypass Garbage Collection freezes
  const floodQueue = new Int32Array(w * h); 
  
  // Flood Fill looking strictly for WHITE/EMPTY areas
  let currentLabel = 1;
  for (let y = 0; y < h; y += 1) { 
      for (let x = 0; x < w; x += 1) {
          const startIdx = y * w + x;
          if (labels[startIdx] === 0) {
              const s4 = startIdx << 2;
              const isMaskWall = maskData[s4 + 3] > 128;
              const threshold = settings.gapThreshold !== undefined ? settings.gapThreshold : 370;
              const isBaseWall = (baseData[s4 + 3] <= 128) || ((baseData[s4] + baseData[s4+1] + baseData[s4+2]) < threshold);

              if (!isMaskWall && !isBaseWall) {
                  floodQueue[0] = startIdx;
                  labels[startIdx] = currentLabel;
                  let count = 0;
                  let sumX = 0, sumY = 0;
                  let touchesBorder = false;
                  
                  let head = 0;
                  let tail = 1;
                  while(head < tail) {
                      const curr = floodQueue[head++];
                      const cx = curr % w;
                      const cy = Math.floor(curr / w);
                      
                      count++;
                      sumX += cx; sumY += cy;
                      
                      if (cx <= 0 || cx >= w-1 || cy <= 0 || cy >= h-1) touchesBorder = true;
                      
                      const neighbors = [curr - 1, curr + 1, curr - w, curr + w];
                      for(let i=0; i<4; i++) {
                          const n = neighbors[i];
                          if (n >= 0 && n < labels.length && labels[n] === 0) {
                              const n4 = n << 2;
                              const nIsMaskWall = maskData[n4 + 3] > 128;
                              const nIsBaseWall = (baseData[n4+3] <= 128) || ((baseData[n4] + baseData[n4+1] + baseData[n4+2]) < (settings.gapThreshold !== undefined ? settings.gapThreshold : 370));

                              if (!nIsMaskWall && !nIsBaseWall) {
                                  labels[n] = currentLabel;
                                  floodQueue[tail++] = n;
                              } else {
                                  labels[n] = -1; // Wall
                              }
                          }
                      }
                  }
                  
                  if (count > 0) {
                      regions.push({
                          id: currentLabel,
                          cx: sumX / count, // Center of Mass (Centroid)
                          cy: sumY / count,
                          count: count,
                          touchesBorder: touchesBorder
                      });
                  }
                  currentLabel++;
              } else {
                  labels[startIdx] = -1; // Wall
              }
          }
      }
  }

  // Fetch pre-calculated spiral offsets instantly
  const offsets = getSpiralOffsets();
  
  // NEW: Topological awareness for custom letters!
  // We check the physical x/y of every moved letter to see which region it lives in.
  const regionHasCustomLetter = new Set();
  const keptCustomLetters = []; // Track which custom letters survive a merge
  
  if (imgObj.hasCustomLetters && imgObj.letters) {
      imgObj.letters.forEach(l => {
          if (!l.isManual) return; // NEW: Ignore untouched letters so they can organically recalculate!
          
          const px = Math.floor(l.x);
          const py = Math.floor(l.y);
          if (px >= 0 && px < w && py >= 0 && py < h) {
              const rId = labels[py * w + px];
              if (rId > 0) {
                  // If region is merged, keep only the FIRST custom letter found, destroy the rest to prevent duplicates
                  if (!regionHasCustomLetter.has(rId)) {
                      regionHasCustomLetter.add(rId);
                      keptCustomLetters.push(l);
                  }
              } else {
                  // Letter is on a wall. Keep it so it isn't deleted, but don't let it claim a region.
                  keptCustomLetters.push(l);
              }
          } else {
              // BUG FIX: The letter was dragged completely off the canvas. Keep it alive anyway!
              keptCustomLetters.push(l);
          }
      });
  }

  // Test Canvas for pixel-perfect letter measurement
  const testCanvas = document.createElement('canvas');
  const testCtx = testCanvas.getContext('2d', { willReadFrequently: true });
  
  const safetyPx = Number(settings.safetyPx) || 0;
  let defaultLow = Number(settings.minSize) || 4;
  let defaultHigh = Number(settings.maxSize) || 35;
  if (defaultLow > defaultHigh) { const t = defaultLow; defaultLow = defaultHigh; defaultHigh = t; }

  const regionsToFill = new Set(); // Memory to track all tiny gaps for batch-filling
  const placedLetters = []; // NEW: Array to store letter objects for manual moving

  for(let r of regions) {
      
      // HARD BLOCK: Completely ignore massive outside background areas to prevent freezing.
      if (r.touchesBorder && r.count > 5000) continue;

      // NEW: If the user manually moved a letter into this specific region, skip auto-generating a new one!
      if (regionHasCustomLetter.has(r.id)) continue;

      // FAST-PATH: If the region is way too tiny to ever fit a letter, skip the expensive math and just instantly black it out.
      // We bumped the fast-path threshold slightly to ensure slightly larger specks are caught.
      if (r.count < 500) {
          if (settings.fillGaps) regionsToFill.add(r.id);
          continue;
      }

      // Abstracted the binary search to support Python's 2-Pass safety fallback
      const tryFit = (margin) => {
          let best = null;
          let low = defaultLow, high = defaultHigh;
          const padding = (margin * 2) + 10;
          
          while(low <= high) {
              let mid = Math.floor((low+high)/2);
              
              testCtx.font = `bold ${mid}px ${settings.font}`;
              testCtx.textBaseline = 'middle';
              testCtx.textAlign = 'center';
              const metrics = testCtx.measureText(settings.char);
              
              // Fallback for browsers that don't fully support actualBoundingBox
              const bw = metrics.actualBoundingBoxRight ? (metrics.actualBoundingBoxRight + metrics.actualBoundingBoxLeft) : testCtx.measureText(settings.char).width;
              const bh = metrics.actualBoundingBoxAscent ? (metrics.actualBoundingBoxAscent + metrics.actualBoundingBoxDescent) : mid;
              
              const fontW = Math.ceil(bw + padding);
              const fontH = Math.ceil(bh + padding);
              
              if (fontW <= 0 || fontH <= 0 || isNaN(fontW) || isNaN(fontH)) break;
              
              testCanvas.width = fontW; testCanvas.height = fontH;
              testCtx.clearRect(0,0,fontW,fontH);
              // Important: Re-assign font properties after width/height change clears them
              testCtx.font = `bold ${mid}px ${settings.font}`;
              testCtx.textBaseline = 'middle';
              testCtx.textAlign = 'center';
              
              // Draw fat stroke for safety bounds
              testCtx.lineJoin = 'round';
              if (margin > 0) {
                  testCtx.lineWidth = margin * 2;
                  testCtx.strokeStyle = 'black';
                  testCtx.strokeText(settings.char, fontW/2, fontH/2);
              }
              testCtx.fillStyle = 'black';
              testCtx.fillText(settings.char, fontW/2, fontH/2);
              
              const lData = testCtx.getImageData(0,0,fontW,fontH).data;
              const letterPixels = [];
              
              // Step by 1 to ensure tiny fonts aren't missed
              for(let ly=0; ly<fontH; ly+=1) {
                  for(let lx=0; lx<fontW; lx+=1) {
                      const idx = (ly * fontW + lx) * 4;
                      if (lData[idx+3] > 128) {
                          letterPixels.push({ x: lx - Math.floor(fontW/2), y: ly - Math.floor(fontH/2) });
                      }
                  }
              }
              
              let foundPos = null;
              for(let o of offsets) {
                  const testX = Math.round(r.cx + o.dx);
                  const testY = Math.round(r.cy + o.dy);
                  
                  if (testX - Math.floor(fontW/2) < 0 || testX + Math.floor(fontW/2) >= w || testY - Math.floor(fontH/2) < 0 || testY + Math.floor(fontH/2) >= h) continue;

                  let valid = true;
                  for(let i=0; i<letterPixels.length; i++) {
                      const px = testX + letterPixels[i].x;
                      const py = testY + letterPixels[i].y;
                      if (labels[py * w + px] !== r.id) {
                          valid = false;
                          break;
                      }
                  }
                  if (valid) { foundPos = { x: testX, y: testY }; break; }
              }
              
              if (foundPos) {
                  best = { size: mid, x: foundPos.x, y: foundPos.y };
                  low = mid + 1; // It fits! Push for a larger size.
              } else {
                  high = mid - 1; // Failed. Drop the size.
              }
          }
          return best;
      };

      // PASS 1: Try requested safety margin
      let bestFit = tryFit(safetyPx);
      
      // PASS 2 (Python Logic): If it fails, fallback to MIN_SAFETY_PX = 2
      if (!bestFit && safetyPx > 2) {
          bestFit = tryFit(2);
      }
      
      // Stamp the absolute best fit
      if (bestFit) {
          let finalSize = bestFit.size;
          // If the best fit is smaller than our readable floor, force it to be bigger
          if (settings.readableFloor && finalSize < settings.readableFloor) {
              finalSize = settings.readableFloor;
          }
          
          ctxL.font = `bold ${finalSize}px ${settings.font}`;
          ctxL.fillStyle = settings.color;
          ctxL.textBaseline = 'middle';
          ctxL.textAlign = 'center';
          ctxL.fillText(settings.char, bestFit.x, bestFit.y);

          // NEW: Save the vector data so we can move it later
          placedLetters.push({
              id: crypto.randomUUID(),
              char: settings.char,
              font: settings.font,
              color: settings.color,
              size: finalSize,
              x: bestFit.x,
              y: bestFit.y
          });
      } else if (settings.fillGaps && r.count < 5000) {
          // Final safety net: If a massive weirdly-shaped region somehow failed to fit a letter,
          // do NOT paint it black. Only fill genuinely small gaps.
          regionsToFill.add(r.id); 
      }
  }
  
  // ⚡ Perform Lightning Fill with 3px Dilation to completely eradicate stubborn anti-aliasing
  const gapsCanvas = document.createElement('canvas');
  gapsCanvas.width = w; gapsCanvas.height = h;
  const ctxG = gapsCanvas.getContext('2d');

  if (regionsToFill.size > 0) {
      const gData = ctxG.getImageData(0, 0, w, h);
      const pxData = gData.data;
      const dilationRadius = 3; 
      
      // O(1) ultra-fast lookup map to completely prevent browser freezing on 8 million pixels
      const fillMap = new Uint8Array(currentLabel + 1);
      regionsToFill.forEach(id => fillMap[id] = 1);
      
      for (let i = 0; i < labels.length; i++) {
          if (fillMap[labels[i]]) {
              const x = i % w;
              const y = Math.floor(i / w);
              
              // Draw a mini 2px square around every filled pixel to swallow thick anti-aliasing
              for (let dy = -dilationRadius; dy <= dilationRadius; dy++) {
                  for (let dx = -dilationRadius; dx <= dilationRadius; dx++) {
                      const nx = x + dx;
                      const ny = y + dy;
                      if (nx >= 0 && nx < w && ny >= 0 && ny < h) {
                          const nIdx = (ny * w + nx) << 2;
                          pxData[nIdx] = pxData[nIdx+1] = pxData[nIdx+2] = 0; 
                          pxData[nIdx+3] = 255;
                      }
                  }
              }
          }
      }
      ctxG.putImageData(gData, 0, 0);
  }
  
  // FORCE GARBAGE COLLECTION: Clean up background measurement canvases and base image
  combinedCanvas.width = 0; combinedCanvas.height = 0;
  testCanvas.width = 0; testCanvas.height = 0;
  baseImg.src = '';

  return { lettersCanvas, gapsCanvas, placedLetters, keptCustomLetters };
};

// --- CRC32 helper for PNG chunk integrity ---
const crc32 = (data) => {
  let crc = 0xFFFFFFFF;
  for (let i = 0; i < data.length; i++) {
    crc ^= data[i];
    for (let j = 0; j < 8; j++) {
      crc = (crc & 1) ? (0xEDB88320 ^ (crc >>> 1)) : (crc >>> 1);
    }
  }
  return (crc ^ 0xFFFFFFFF) >>> 0;
};

// --- Injects pHYs DPI metadata chunk into a PNG blob ---
const injectPngDpi = async (blob, dpi) => {
  const buffer = await blob.arrayBuffer();
  const bytes = new Uint8Array(buffer);
  const ppm = Math.round(dpi / 0.0254);
  const phys = new Uint8Array(21);
  phys[0]=0; phys[1]=0; phys[2]=0; phys[3]=9;
  phys[4]=112; phys[5]=72; phys[6]=89; phys[7]=115;
  phys[8]=(ppm>>24)&255; phys[9]=(ppm>>16)&255; phys[10]=(ppm>>8)&255; phys[11]=ppm&255;
  phys[12]=(ppm>>24)&255; phys[13]=(ppm>>16)&255; phys[14]=(ppm>>8)&255; phys[15]=ppm&255;
  phys[16]=1;
  const crcData = phys.slice(4, 17);
  const crc = crc32(crcData);
  phys[17]=(crc>>24)&255; phys[18]=(crc>>16)&255; phys[19]=(crc>>8)&255; phys[20]=crc&255;
  const result = new Uint8Array(bytes.length + 21);
  result.set(bytes.slice(0, 33), 0);
  result.set(phys, 33);
  result.set(bytes.slice(33), 54);
  return new Blob([result], { type: 'image/png' });
};

// --- NATIVE LOCAL DATABASE (INDEXEDDB) ---
const DB_NAME = 'MonoSegmentDB';
const STORE_NAME = 'projects';

const initDB = () => new Promise((resolve, reject) => {
  const request = indexedDB.open(DB_NAME, 1);
  request.onerror = () => reject(request.error);
  request.onsuccess = () => resolve(request.result);
  request.onupgradeneeded = (e) => e.target.result.createObjectStore(STORE_NAME);
});
const localSet = async (key, val) => { const db = await initDB(); return new Promise((res, rej) => { const tx = db.transaction(STORE_NAME, 'readwrite'); tx.objectStore(STORE_NAME).put(val, key); tx.oncomplete = res; tx.onerror = () => rej(tx.error); }); };
const localGet = async (key) => { const db = await initDB(); return new Promise((res, rej) => { const tx = db.transaction(STORE_NAME, 'readonly'); const req = tx.objectStore(STORE_NAME).get(key); req.onsuccess = () => res(req.result); req.onerror = () => rej(req.error); }); };
const localDel = async (key) => { const db = await initDB(); return new Promise((res, rej) => { const tx = db.transaction(STORE_NAME, 'readwrite'); tx.objectStore(STORE_NAME).delete(key); tx.oncomplete = res; tx.onerror = () => rej(tx.error); }); };

// --- CACHE MANAGER HELPERS ---
const getLocalStats = async () => { const db = await initDB(); return new Promise((res, rej) => { const tx = db.transaction(STORE_NAME, 'readonly'); const req = tx.objectStore(STORE_NAME).getAllKeys(); req.onsuccess = () => { let projects = 0; let assets = 0; req.result.forEach(k => k.startsWith('asset_') ? assets++ : projects++); res({ projects, assets }); }; req.onerror = () => rej(tx.error); }); };
const clearLocalCache = async () => { const db = await initDB(); return new Promise((res, rej) => { const tx = db.transaction(STORE_NAME, 'readwrite'); const req = tx.objectStore(STORE_NAME).clear(); req.onsuccess = res; req.onerror = () => rej(tx.error); }); };

// --- ASSET-SEPARATED DUAL STORAGE ENGINE ---
const CHUNK_SIZE = 150000; // Ultra-safe chunk size (150kb) to prevent HTTP 400 errors
const assetQueue = [];
let isProcessingQueue = false;

const StorageEngine = {
  onSyncChange: null,
  onAssetSuccess: null,
  queuedAssetIds: new Set(),

  saveState: async (key, stateObj, userParam) => {
    const payloadStr = JSON.stringify(stateObj);
    await localSet(key, payloadStr);
    const user = userParam || auth?.currentUser;
    if (db && user) {
       try { await setDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'projects', key), { payload: payloadStr, timestamp: Date.now() }); } catch (err) {}
    }
  },
  loadState: async (key, userParam) => {
    let payloadStr = null;
    const user = userParam || auth?.currentUser;
    if (db && user) {
       try {
          const docSnap = await getDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'projects', key));
          if (docSnap.exists()) payloadStr = docSnap.data().payload;
       } catch (err) {}
    }
    if (!payloadStr) payloadStr = await localGet(key);
    return payloadStr ? JSON.parse(payloadStr) : null;
  },
  deleteState: async (key, userParam) => {
    await localDel(key);
    const user = userParam || auth?.currentUser;
    if (db && user) { try { await deleteDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'projects', key)); } catch (e) {} }
  },

  // Throttled Heavy Assets Engine
  queueAssetSave: (id, base64Str, userParam) => {
    if (StorageEngine.queuedAssetIds.has(id)) return;
    StorageEngine.queuedAssetIds.add(id);
    assetQueue.push({ id, base64Str, userParam });
    StorageEngine.processQueue();
  },

  processQueue: async () => {
    if (isProcessingQueue || assetQueue.length === 0) return;
    isProcessingQueue = true;
    if (StorageEngine.onSyncChange) StorageEngine.onSyncChange(true);

    while (assetQueue.length > 0) {
        const { id, base64Str, userParam } = assetQueue[0];
        const user = userParam || auth?.currentUser;
        
        await localSet(`asset_${id}`, base64Str);

        if (db && user) {
            try {
                const numChunks = Math.ceil(base64Str.length / CHUNK_SIZE);
                for (let i = 0; i < numChunks; i++) {
                    const chunkStr = base64Str.slice(i * CHUNK_SIZE, (i + 1) * CHUNK_SIZE);
                    await setDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'assets', `${id}_chunk_${i}`), { payload: chunkStr });
                    // THROTTLE: Give Firebase 50ms breathing room between chunks to prevent Queue Exhaustion
                    await new Promise(r => setTimeout(r, 50)); 
                }
                await setDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'assets', id), { chunks: numChunks });
                
                if (StorageEngine.onAssetSuccess) StorageEngine.onAssetSuccess(id);
            } catch (err) { 
                console.warn("Asset upload failed, will retry next load:", err); 
                StorageEngine.queuedAssetIds.delete(id); // Allow retry later
            }
        }
        assetQueue.shift(); // Remove from queue and move to the next image
    }

    isProcessingQueue = false;
    if (StorageEngine.onSyncChange) StorageEngine.onSyncChange(false);
  },
  
  loadAsset: async (id, userParam) => {
    let base64Str = null;
    const user = userParam || auth?.currentUser;
    if (db && user) {
       try {
          const metaSnap = await getDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'assets', id));
          if (metaSnap.exists()) {
              const numChunks = metaSnap.data().chunks;
              const chunkPromises = [];
              for (let i = 0; i < numChunks; i++) {
                  chunkPromises.push(getDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'assets', `${id}_chunk_${i}`)));
              }
              const chunkSnaps = await Promise.all(chunkPromises);
              base64Str = chunkSnaps.map(snap => snap.exists() ? snap.data().payload : '').join('');
          }
       } catch (err) { console.warn("Cloud asset load failed:", err); }
    }
    if (!base64Str) base64Str = await localGet(`asset_${id}`);
    return base64Str;
  },
  deleteAsset: async (id, userParam) => {
    // Intercept and remove from queue if it hasn't uploaded yet
    const qIndex = assetQueue.findIndex(item => item.id === id);
    if (qIndex > -1) assetQueue.splice(qIndex, 1);
    StorageEngine.queuedAssetIds.delete(id);

    await localDel(`asset_${id}`);
    const user = userParam || auth?.currentUser;
    if (db && user) {
       try {
           const metaSnap = await getDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'assets', id));
           if (metaSnap.exists()) {
               for (let i = 0; i < metaSnap.data().chunks; i++) deleteDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'assets', `${id}_chunk_${i}`));
           }
           await deleteDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'assets', id));
       } catch (e) {}
    }
  }
};

// Prevents data loss if user tries to close tab while heavy images are syncing in the background

      e.preventDefault(); // Standard requirement for modern browsers
      e.returnValue = 'Images are still syncing to the cloud. If you leave now, recent uploads will not be saved.'; 
      return e.returnValue;
  }
});

      const isInput = e.target.tagName === 'INPUT' || e.target.tagName === 'TEXTAREA' || e.target.tagName === 'SELECT';

      if (viewerMode === 'edit') {
        const ctrlOrCmd = e.ctrlKey || e.metaKey;

        // Instant Magic-Erase Quick Toggle (Hold Alt)
        if (e.key === 'Alt') {
          e.preventDefault();
          if (!preAltToolRef.current) {
             // 1. Force kill any mid-stroke drawing to prevent wild glitches
             setIsDrawing(false);
             isDraggingHandle.current = false;
             drawPoints.current = [];
             setPreviewPos(null);
             
             // 2. Save current tool to memory, then engage Magic Eraser
             preAltToolRef.current = { tool: editTool, mode: eraserMode };
             setEditTool('erase');
             setEraserMode('magic');
          }
          return;
        }

        if (ctrlOrCmd && e.key.toLowerCase() === 'z') {
          e.preventDefault();
          if (e.shiftKey) handleRedo();
          else handleUndo();
          return;
        } else if (ctrlOrCmd && e.key.toLowerCase() === 'y') {
          e.preventDefault();
          handleRedo();
          return;
        }

        if (ctrlOrCmd && e.key === 'Enter') {
          e.preventDefault();
          document.getElementById('apply-btn')?.click();
          return;
        }

        if (!isInput && !ctrlOrCmd && (e.key === 'Escape')) {
          e.preventDefault();
          // Route the Escape key through the official Cancel function to safely restore the snapshot
          cancelEdits(); 
          return;
        }
        if (!isInput && !ctrlOrCmd && (e.key === 'Enter')) {
          if (editTool === 'pen') {
            e.preventDefault();
            document.getElementById('finish-path-btn')?.click();
          }
          return;
        }
        if (!isInput && !ctrlOrCmd) {
          if (e.key.toLowerCase() === 'b') {
            e.preventDefault();
            setEditTool('brush');
            setPenNodes([]);
            setPreviewPos(null);
            return;
          }
          if (e.key.toLowerCase() === 'e') {
            e.preventDefault();
            setEditTool('erase');
            setPenNodes([]);
            setPreviewPos(null);
            return;
          }
          if (e.key.toLowerCase() === 'g') {
            e.preventDefault();
            setEditTool('gapBrush');
            setPenNodes([]);
            setPreviewPos(null);
            return;
          }
          if (e.key.toLowerCase() === 'p') {
            e.preventDefault();
            setEditTool('pen');
            setPenNodes([]);
            setPreviewPos(null);
            return;
          }
        }

      }

      if (!isInput && e.key.toLowerCase() === 'r' && viewerMode !== 'edit' && imagesRef.current.length > 0 && activeIndexRef.current >= 0) {
        e.preventDefault();
        setViewerMode('edit');
        return;
      }

      if (e.code === 'Space' && !isInput) {
        e.preventDefault(); 
        setIsSpacePressed(true);
      }

      // --- Nudge Letters or Navigate Filmstrip ---
      if (!isInput) {
        if (viewerMode === 'edit' && editTool === 'moveLetter' && selectedLetterId) {
          e.preventDefault();
          const step = e.shiftKey ? 10 : 1; // Shift + Arrow moves faster
          setImages(prev => prev.map((img, i) => {
             if (i !== activeIndexRef.current || !img.letters) return img;
             const newLetters = img.letters.map(l => {
                 if (l.id !== selectedLetterId) return l;
                 let nx = l.x, ny = l.y;
                 if (e.key === 'ArrowUp') ny -= step;
                 if (e.key === 'ArrowDown') ny += step;
                 if (e.key === 'ArrowLeft') nx -= step;
                 if (e.key === 'ArrowRight') nx += step;
                 return { ...l, x: nx, y: ny, isManual: true }; // TAG AS MANUAL
             });
             return { ...img, letters: newLetters, hasCustomLetters: true };
          }));
          return;
        }

        // --- Resize Individual Letters via Keyboard ---
        if (viewerMode === 'edit' && editTool === 'moveLetter' && selectedLetterId) {
            if (e.key === '-' || e.key === '_' || e.key === '[') {
                 e.preventDefault();
                 setImages(prev => prev.map((img, i) => {
                     if (i !== activeIndexRef.current || !img.letters) return img;
                     const newLetters = img.letters.map(l => l.id === selectedLetterId ? { ...l, size: Math.max(1, l.size - (e.shiftKey ? 5 : 1)), isManual: true } : l);
                     return { ...img, letters: newLetters, hasCustomLetters: true };
                 }));
                 return;
            }
            if (e.key === '=' || e.key === '+' || e.key === ']') {
                 e.preventDefault();
                 setImages(prev => prev.map((img, i) => {
                     if (i !== activeIndexRef.current || !img.letters) return img;
                     const newLetters = img.letters.map(l => l.id === selectedLetterId ? { ...l, size: l.size + (e.shiftKey ? 5 : 1), isManual: true } : l);
                     return { ...img, letters: newLetters, hasCustomLetters: true };
                 }));
                 return;
            }
        }

        if (imagesRef.current.length > 1) {
          if (e.key === 'ArrowLeft') {
            e.preventDefault();
            if (viewerMode === 'edit') return; // Lock navigation until Apply/Cancel
            const currentIdx = activeIndexRef.current;
            if (currentIdx > 0) setActiveIndex(currentIdx - 1);
          } else if (e.key === 'ArrowRight') {
            e.preventDefault();
            if (viewerMode === 'edit') return; // Lock navigation until Apply/Cancel
            const currentIdx = activeIndexRef.current;
            if (currentIdx < imagesRef.current.length - 1) setActiveIndex(currentIdx + 1);
          }
        }
      }
    };
    
    const handleKeyUp = (e) => {
      if (e.code === 'Space') {
        setIsSpacePressed(false);
        setIsPanning(false);
      }
      if (e.key === 'Alt' && preAltToolRef.current) {
        setEditTool(preAltToolRef.current.tool);
        setEraserMode(preAltToolRef.current.mode);
        preAltToolRef.current = null;
      }
    };
    
    const handleBlur = () => {
      setIsSpacePressed(false);
      setIsPanning(false);
      // Safety net: if they Alt+Tab away from the window, release the tool
      if (preAltToolRef.current) {
        setEditTool(preAltToolRef.current.tool);
        setEraserMode(preAltToolRef.current.mode);
        preAltToolRef.current = null;
      }
    };
    
    window.addEventListener('keydown', handleKeyDown);
    window.addEventListener('keyup', handleKeyUp);
    window.addEventListener('blur', handleBlur);
    return () => {
      window.removeEventListener('keydown', handleKeyDown);
      window.removeEventListener('keyup', handleKeyUp);
      window.removeEventListener('blur', handleBlur);
    }
  }, [viewerMode, editTool, eraserMode, selectedLetterId]); // FIX: Added selectedLetterId so the keyboard knows what you clicked!

  useEffect(() => {
    const handleGlobalMouseMove = (e) => {
      if (isPanning) {
         const dx = e.clientX - panStart.current.x;
         const dy = e.clientY - panStart.current.y;
         if (workspaceRef.current) {
           workspaceRef.current.scrollLeft = panStart.current.sl - dx;
           workspaceRef.current.scrollTop = panStart.current.st - dy;
         }
      }
    };
    const handleGlobalMouseUp = () => {
      if (isPanning) setIsPanning(false);
    };

    if (isPanning) {
      window.addEventListener('mousemove', handleGlobalMouseMove);
      window.addEventListener('mouseup', handleGlobalMouseUp);
    }
    return () => {
      window.removeEventListener('mousemove', handleGlobalMouseMove);
      window.removeEventListener('mouseup', handleGlobalMouseUp);
    };
  }, [isPanning]);

  const handleWorkspaceMouseDown = (e) => {
    if (e.button === 1 || (e.button === 0 && isSpacePressed)) {
      e.preventDefault();
      setIsPanning(true);
      panStart.current = { 
        x: e.clientX, 
        y: e.clientY, 
        sl: workspaceRef.current.scrollLeft, 
        st: workspaceRef.current.scrollTop 
      };
    }
  };

  const updateActiveImage = (updates) => {
    setImages(prev => {
      const newImgs = [...prev];
      const idx = activeIndexRef.current;
      if (idx >= 0 && idx < newImgs.length) {
        newImgs[idx] = { ...newImgs[idx], ...updates };
      }
      return newImgs;
    });
  };

  const updateHistoryState = () => {
    setCanUndo(historyIndexRef.current > 0);
    setCanRedo(historyIndexRef.current < historyRef.current.length - 1);
  };

  const updatePathsAndHistory = (newPaths) => {
    const newHistory = historyRef.current.slice(0, historyIndexRef.current + 1);
    newHistory.push(newPaths);
    
    if (newHistory.length > 50) newHistory.shift();
    
    historyRef.current = newHistory;
    historyIndexRef.current = newHistory.length - 1;
    updateActiveImage({ paths: newPaths, history: newHistory, historyIndex: historyIndexRef.current });
    updateHistoryState();
  };

  // Helper to sync UI sliders to the current image's parameters instantly
  const updateActiveParams = (key, val) => {
    if (key === 'thick') setLineThickness(val);
    if (key === 'minCut') setMinCutLength(val);
    if (key === 'maxCut') setMaxCutLength(val);
    if (key === 'spacing') setMinSpacing(val);
    if (key === 'smoothness') setCurveSmoothness(val);

    if (activeImg && activeImg.params) {
        updateActiveImage({
            params: { ...activeImg.params, [key]: val }
        });
    }
  };

  const handleUndo = () => {
    if (historyIndexRef.current > 0) {
      historyIndexRef.current -= 1;
      const revertedPaths = historyRef.current[historyIndexRef.current];
      updateActiveImage({ paths: revertedPaths, historyIndex: historyIndexRef.current });
      updateHistoryState();
      setPenNodes([]); 
      setPreviewPos(null);
    }
  };

  const handleRedo = () => {
    if (historyIndexRef.current < historyRef.current.length - 1) {
      historyIndexRef.current += 1;
      const forwardedPaths = historyRef.current[historyIndexRef.current];
      updateActiveImage({ paths: forwardedPaths, historyIndex: historyIndexRef.current });
      updateHistoryState();
      setPenNodes([]); 
      setPreviewPos(null);
    }
  };

  // --- Custom Font Uploader ---
  const handleCustomFontUpload = async (e) => {
    const file = e.target.files?.[0];
    if (!file) return;
    
    try {
      const fontName = 'CustomFont_' + Date.now();
      const buffer = await file.arrayBuffer();
      const font = new FontFace(fontName, buffer);
      await font.load();
      document.fonts.add(font);
      handleLetterSettingChange('font', `"${fontName}"`);
    } catch (err) {
      console.error("Failed to load custom font", err);
      showToast("Failed to load font file.");
    }
    e.target.value = '';
  };

  // --- Helper: Generate Blobs from Image Object (Including Letter Placing Pipeline) ---
  const regenerateBlobsForImage = async (imgObj, pathsOverride, lettersOverride) => {
    const pathsToUse = pathsOverride || imgObj.paths;
    const currentLS = letterSettingsRef.current;
    const w = imgObj.w; const h = imgObj.h;
    
    const img = new Image();
    img.src = imgObj.dataUrl;
    await new Promise(r => { img.onload = r; });

    // Separate paths into "Lines" and "Blockouts"
    const linePaths = pathsToUse.filter(p => !p.isGapFill);
    const blockoutPaths = pathsToUse.filter(p => p.isGapFill);

    // Build the Blockout Layer
    const blockoutCanvas = document.createElement('canvas');
    blockoutCanvas.width = w; blockoutCanvas.height = h;
    const bCtx = blockoutCanvas.getContext('2d');
    renderAllPaths(bCtx, w, h, blockoutPaths);
    
    // FIX: Clip the manual blockouts to the original image so it doesn't draw in transparent voids
    bCtx.globalCompositeOperation = 'destination-in';
    bCtx.drawImage(img, 0, 0);

    // Create the Combined Silhouette (Base Image + Manual Blockouts)
    const silhouetteCanvas = document.createElement('canvas');
    silhouetteCanvas.width = w; silhouetteCanvas.height = h;
    const silCtx = silhouetteCanvas.getContext('2d');
    silCtx.drawImage(img, 0, 0);
    silCtx.drawImage(blockoutCanvas, 0, 0);

    // Build the Topo Mask (Lines Only)
    const maskCanvas = document.createElement('canvas');
    maskCanvas.width = w; maskCanvas.height = h;
    const masterCtx = maskCanvas.getContext('2d');
    
    renderAllPaths(masterCtx, w, h, linePaths);
    masterCtx.globalCompositeOperation = 'destination-in';
    // Clip the lines against the combined silhouette so lines can traverse blockouts
    masterCtx.drawImage(silhouetteCanvas, 0, 0); 
    
    let lettersUrl = null;
    let lCanvas = null;

    // Create the Final Base Canvas (will hold Original + Manual Blockouts + Auto Gaps)
    const finalBaseCanvas = document.createElement('canvas');
    finalBaseCanvas.width = w; finalBaseCanvas.height = h;
    const finalBaseCtx = finalBaseCanvas.getContext('2d');
    finalBaseCtx.drawImage(silhouetteCanvas, 0, 0);

    let finalLetters = [];

    // STEP 2: Execute Letter Placer if Enabled
    if (currentLS.enabled) {
        const result = await placeLetters(maskCanvas, imgObj, currentLS);
        const gCanvas = result.gapsCanvas;
        
        // Decide whether to use user-moved letters or the auto-generated ones
        if (imgObj.hasCustomLetters) {
            // Apply any new global style settings (color, font) to the SURVIVING custom letters
            const updatedCustomLetters = result.keptCustomLetters.map(l => ({
                ...l,
                char: currentLS.char,
                font: currentLS.font,
                color: currentLS.color
            }));
            // Merge them! The surviving custom letters stay, and completely new regions get new letters!
            finalLetters = [...updatedCustomLetters, ...result.placedLetters];
        } else {
            finalLetters = result.placedLetters;
        }
        
        // Manually build the letters canvas so we can respect custom positions
        lCanvas = document.createElement('canvas');
        lCanvas.width = w; lCanvas.height = h;
        const ctxL = lCanvas.getContext('2d');
        finalLetters.forEach(letObj => {
            ctxL.font = `bold ${letObj.size}px ${letObj.font}`;
            ctxL.fillStyle = letObj.color;
            ctxL.textBaseline = 'middle';
            ctxL.textAlign = 'center';
            ctxL.fillText(letObj.char, letObj.x, letObj.y);
        });
        
        // FIX: Clip the auto-gaps to the silhouette so the 3px dilation doesn't bleed into transparent voids
        const gCtx = gCanvas.getContext('2d');
        gCtx.globalCompositeOperation = 'destination-in';
        gCtx.drawImage(silhouetteCanvas, 0, 0);
        
        // Add the Auto-Gaps directly to the Final Base Art (not the letters canvas)
        finalBaseCtx.drawImage(gCanvas, 0, 0);
        
        const lettersBlob = await injectPngDpi(await new Promise(res => lCanvas.toBlob(res, 'image/png')), imgObj.dpi || 300);
        lettersUrl = URL.createObjectURL(lettersBlob);
        
        gCanvas.width = 0; gCanvas.height = 0; // Clean up memory
    }

    // Build the Full Art Base (Starts with the Final Base Art including all gaps/blockouts)
    const fullCanvas = document.createElement('canvas');
    fullCanvas.width = w; fullCanvas.height = h;
    const fCtx = fullCanvas.getContext('2d');
    fCtx.drawImage(finalBaseCanvas, 0, 0);
    
    // Apply lines to the Full Art Base ONLY if hideLines is toggled off
    if (!hideLinesRef.current) {
        fCtx.globalCompositeOperation = lineBlendModeRef.current === 'multiply' ? 'multiply' : 'source-atop';
        fCtx.drawImage(maskCanvas, 0, 0);
        
        // If we used multiply, we must clip off any line overflow from transparent background areas
        if (lineBlendModeRef.current === 'multiply') {
            fCtx.globalCompositeOperation = 'destination-in';
            // Clip the final art to the shape of the modified base art
            fCtx.drawImage(finalBaseCanvas, 0, 0);
        }
    }
    
    // Apply Letters on Top
    if (lCanvas) {
        fCtx.globalCompositeOperation = currentLS.blendMode === 'normal' ? 'source-over' : (currentLS.blendMode || 'multiply');
        fCtx.drawImage(lCanvas, 0, 0);
        
        // If we used multiply, we must clip off any letter overflow from transparent background areas
        if ((currentLS.blendMode || 'multiply') === 'multiply') {
            fCtx.globalCompositeOperation = 'destination-in';
            fCtx.drawImage(finalBaseCanvas, 0, 0);
        }
        
        lCanvas.width = 0; lCanvas.height = 0;
    }

    // Parallelize blob generation to utilize multiple browser threads simultaneously
    const [fullBlob, maskBlob, baseBlob] = await Promise.all([
        new Promise(res => fullCanvas.toBlob(res, 'image/png')).then(b => injectPngDpi(b, imgObj.dpi || 300)),
        new Promise(res => maskCanvas.toBlob(res, 'image/png')).then(b => injectPngDpi(b, imgObj.dpi || 300)),
        new Promise(res => finalBaseCanvas.toBlob(res, 'image/png')).then(b => injectPngDpi(b, imgObj.dpi || 300))
    ]);

    const oldImage = imgObj.resultImage;
    const oldMask = imgObj.resultMaskURL;
    const oldLetters = imgObj.resultLettersURL;
    const oldBase = imgObj.resultBaseURL;

    // Defer revocation so React has time to render the new blob URLs without flashing a broken image
    setTimeout(() => {
        if (oldImage) URL.revokeObjectURL(oldImage);
        if (oldMask) URL.revokeObjectURL(oldMask);
        if (oldLetters) URL.revokeObjectURL(oldLetters);
        if (oldBase) URL.revokeObjectURL(oldBase);
    }, 2000);

    // FORCE GARBAGE COLLECTION: Instantly free the memory layer canvases & source image
    maskCanvas.width = 0; maskCanvas.height = 0;
    fullCanvas.width = 0; fullCanvas.height = 0;
    silhouetteCanvas.width = 0; silhouetteCanvas.height = 0;
    blockoutCanvas.width = 0; blockoutCanvas.height = 0;
    finalBaseCanvas.width = 0; finalBaseCanvas.height = 0;
    img.src = '';

    return {
        resultImage: URL.createObjectURL(fullBlob),
        resultMaskURL: URL.createObjectURL(maskBlob),
        resultLettersURL: lettersUrl,
        resultBaseURL: URL.createObjectURL(baseBlob),
        letters: finalLetters // Save the final coordinates back to the state
    };
  };

  // --- Trigger Global Retooling on Letter Setting Change ---
  const handleLetterSettingChange = (key, value) => {
    setLetterSettings(prev => ({ ...prev, [key]: value }));
    
    if (colorChangeTimeoutRef.current) clearTimeout(colorChangeTimeoutRef.current);
    colorChangeTimeoutRef.current = setTimeout(async () => {
        setIsProcessing(true);
        const currentImgs = imagesRef.current; 
        for (let i = 0; i < currentImgs.length; i++) {
            const imgObj = currentImgs[i];
            if (imgObj && imgObj.resultImage) {
                const newBlobs = await regenerateBlobsForImage(imgObj, imgObj.paths);
                // Safe functional state update to prevent stale data overwrites
                setImages(prev => prev.map((img, idx) => idx === i ? { ...img, ...newBlobs } : img));
            }
        }
        setIsProcessing(false);
    }, 600); // Debounce to allow user to finish typing/sliding
  };

  const handleLineColorChange = (newColor) => {
    setLineColor(newColor);
    if (!/^#[0-9A-Fa-f]{6}$/i.test(newColor)) return;
    if (imagesRef.current.length === 0) return;

    // 1. Synchronously update vector paths in state for instant raw data tracking, and update the params color
    setImages(prev => prev.map((img) => {
        if (!img.paths || img.paths.length === 0) return img;
        return {
            ...img,
            params: { ...img.params, color: newColor },
            paths: img.paths.map(p => ({ ...p, color: p.isEraser ? null : (p.isGapFill ? '#000000' : newColor) })),
            history: img.history.map(h => h.map(p => ({ ...p, color: p.isEraser ? null : (p.isGapFill ? '#000000' : newColor) })))
        };
    }));

    const activeIdx = activeIndexRef.current;
    if (activeIdx >= 0 && imagesRef.current[activeIdx] && imagesRef.current[activeIdx].paths) {
        historyRef.current = imagesRef.current[activeIdx].history.map(h => h.map(p => ({ ...p, color: p.isEraser ? null : (p.isGapFill ? '#000000' : newColor) })));
    }

    // 2. Debounce rasterization to eliminate race conditions and browser freezing
    if (colorChangeTimeoutRef.current) clearTimeout(colorChangeTimeoutRef.current);
    colorChangeTimeoutRef.current = setTimeout(async () => {
        const nextImages = [...imagesRef.current]; 
        let bgUpdated = false;
        
        // Prioritize actively viewed image
        if (activeIdx >= 0 && nextImages[activeIdx] && nextImages[activeIdx].resultImage) {
            const imgObj = nextImages[activeIdx];
            const recoloredPaths = imgObj.paths.map(p => ({ ...p, color: p.isEraser ? null : (p.isGapFill ? '#000000' : newColor) }));
            const newBlobs = await regenerateBlobsForImage(imgObj, recoloredPaths);
            
            setImages(prev => prev.map((img, idx) => idx === activeIdx ? { ...img, ...newBlobs } : img));
        }

        // Process the rest of the batch silently in background
        for (let i = 0; i < nextImages.length; i++) {
            if (i === activeIdx) continue; 
            const imgObj = nextImages[i];
            if (imgObj && imgObj.resultImage) {
                const recoloredPaths = imgObj.paths.map(p => ({ ...p, color: p.isEraser ? null : (p.isGapFill ? '#000000' : newColor) }));
                const newBlobs = await regenerateBlobsForImage(imgObj, recoloredPaths);
                setImages(prev => prev.map((img, idx) => idx === i ? { ...img, ...newBlobs } : img));
                bgUpdated = true;
            }
        }
    }, 150); 
  };

  const handleBlendModeChange = (mode) => {
    setLineBlendMode(mode);
    lineBlendModeRef.current = mode;
    
    if (colorChangeTimeoutRef.current) clearTimeout(colorChangeTimeoutRef.current);
    colorChangeTimeoutRef.current = setTimeout(async () => {
        setIsProcessing(true);
        const currentImgs = imagesRef.current; 
        const activeIdx = activeIndexRef.current;
        
        if (activeIdx >= 0 && currentImgs[activeIdx] && currentImgs[activeIdx].resultImage) {
            const imgObj = currentImgs[activeIdx];
            const newBlobs = await regenerateBlobsForImage(imgObj, imgObj.paths);
            setImages(prev => prev.map((img, idx) => idx === activeIdx ? { ...img, ...newBlobs } : img));
        }

        for (let i = 0; i < currentImgs.length; i++) {
            if (i === activeIdx) continue; 
            const imgObj = currentImgs[i];
            if (imgObj && imgObj.resultImage) {
                const newBlobs = await regenerateBlobsForImage(imgObj, imgObj.paths);
                setImages(prev => prev.map((img, idx) => idx === i ? { ...img, ...newBlobs } : img));
            }
        }
        setIsProcessing(false);
    }, 150);
  };

  const toggleHideLines = () => {
    const newVal = !hideLines;
    setHideLines(newVal);
    hideLinesRef.current = newVal; 
    
    if (colorChangeTimeoutRef.current) clearTimeout(colorChangeTimeoutRef.current);
    colorChangeTimeoutRef.current = setTimeout(async () => {
        setIsProcessing(true);
        const currentImgs = imagesRef.current; 
        for (let i = 0; i < currentImgs.length; i++) {
            const imgObj = currentImgs[i];
            if (imgObj && imgObj.resultImage) {
                const newBlobs = await regenerateBlobsForImage(imgObj, imgObj.paths);
                setImages(prev => prev.map((img, idx) => idx === i ? { ...img, ...newBlobs } : img));
            }
        }
        setIsProcessing(false);
    }, 150);
  };

  // --- Handlers: File Upload & Multi-Image Mgmt ---
  const handleFileUpload = (event) => {
    const files = Array.from(event.target.files).filter(f => f.type.startsWith('image/'));
    if (files.length > 0) processFiles(files);
    event.target.value = ''; 
  };

  const handleDrop = (event) => {
    event.preventDefault();
    const files = Array.from(event.dataTransfer.files).filter(f => f.type.startsWith('image/'));
    if (files.length > 0) processFiles(files);
  };

  const processFiles = async (files) => {
    setIsProcessing(true);
    const newImages = [];
    
    for (let file of files) {
        let dataUrl;
        let dpi = 72;
        let imgW, imgH;

        if (file.type === 'image/svg+xml' || file.name.toLowerCase().endsWith('.svg')) {
            try {
                const text = await file.text();
                const parser = new DOMParser();
                const svgDoc = parser.parseFromString(text, "image/svg+xml");
                const svgEl = svgDoc.documentElement;

                const targetW = svgCanvasSize === 'custom' ? (Number(svgCustomW) || 2465) : parseInt(svgCanvasSize.split('x')[0], 10);
                const targetH = svgCanvasSize === 'custom' ? (Number(svgCustomH) || 3300) : parseInt(svgCanvasSize.split('x')[1], 10);

                svgEl.setAttribute('width', targetW + 'px');
                svgEl.setAttribute('height', targetH + 'px');

                if (svgScaleMode === 'stretch') {
                    svgEl.setAttribute('preserveAspectRatio', 'none');
                } else {
                    svgEl.setAttribute('preserveAspectRatio', 'xMidYMid meet');
                }

                const allElements = svgEl.querySelectorAll('*');
                allElements.forEach(el => {
                   if (el.tagName && el.tagName.toLowerCase() !== 'style') {
                       el.removeAttribute('fill');
                       el.removeAttribute('stroke');
                       el.removeAttribute('stroke-width');
                       el.removeAttribute('style'); 
                   }
                });

                const styleFill = svgDoc.createElementNS('http://www.w3.org/2000/svg', 'style');
                styleFill.textContent = `
                  path, rect, circle, ellipse, polygon, polyline, line, text, tspan, use {

  // Helper logic extracted for scanning a single image's optimal parameters
  const calibrateImage = async (imgObj) => {
    const img = new Image();
    img.src = imgObj.dataUrl;
    await new Promise((r) => { img.onload = r; });
    const w = img.width; const h = img.height;

  // --- 🧠 TOPOLOGICALLY AWARE PROCESSING CORE ---
  const runAwareSegmenterCore = async (imgObj, params) => {
    const startTime = performance.now();
    await new Promise(resolve => setTimeout(resolve, 50)); 
    
    // Pre-calculate expensive Trigonometry to prevent millions of CPU math operations
    const trigCache = [];
    for (let angle = 0; angle < 180; angle += 10) {
      trigCache.push({ dx: Math.cos(angle * Math.PI / 180), dy: Math.sin(angle * Math.PI / 180), angle });
    } 

    const thick = Number(params.thick) || 1;
    const minCut = Number(params.minCut) || 50;
    const maxCut = Number(params.maxCut) || 200;
    const spacing = Number(params.spacing) || 100;
    const smoothness = Number(params.smoothness) || 0;
    const color = params.color || '#000000';
    
    const w = imgObj.w; 
    const h = imgObj.h;
    
    const img = new Image();
    img.src = imgObj.dataUrl;
    await new Promise((resolve) => { img.onload = resolve; });

    const canvas = document.createElement('canvas');
    canvas.width = w; canvas.height = h;
    const ctx = canvas.getContext('2d', { willReadFrequently: true });
    ctx.drawImage(img, 0, 0);

    const maskCanvas = document.createElement('canvas');
    maskCanvas.width = w; maskCanvas.height = h;
    const maskCtx = maskCanvas.getContext('2d');

    const data = ctx.getImageData(0, 0, w, h).data;

    const isWall = (x, y) => {
      const ix = x | 0; const iy = y | 0; // Bitwise OR is 20x faster than Math.floor()
      if (ix < 0 || ix >= w || iy < 0 || iy >= h) return true;
      const idx = (iy * w + ix) << 2; // Bitwise shift is faster than multiply
      if (data[idx + 3] < 128) return true; 
      // Skip division entirely, just check if sum is less than 600 (200 * 3)
      return (data[idx] + data[idx+1] + data[idx+2]) < 600; 
    };

    const castRay = (cx, cy, dx, dy, maxDist) => {
      let x = cx; let y = cy;
      for (let i = 0; i < maxDist; i++) {
        if (isWall(x, y)) return { dist: i, p: { x: Math.floor(x), y: Math.floor(y) } };
        x += dx; y += dy;
      }
      return { dist: maxDist, p: null };
    };

    const getInwardNormal = (x, y, originX, originY) => {
      let nx = 0, ny = 0; const radius = 4;
      for(let oy = -radius; oy <= radius; oy++) {
          for(let ox = -radius; ox <= radius; ox++) {
              if(ox === 0 && oy === 0) continue;
              if(isWall(x + ox, y + oy)) { nx -= ox; ny -= oy; }
          }
      }
      const mag = Math.sqrt(nx * nx + ny * ny);
      if (mag > 0.1) {
          nx /= mag; ny /= mag;
          const dirX = originX - x, dirY = originY - y;
          if (nx * dirX + ny * dirY < 0) { nx = -nx; ny = -ny; }
          return {x: nx, y: ny};
      }
      const dirX = originX - x, dirY = originY - y;
      const dMag = Math.sqrt(dirX * dirX + dirY * dirY);
      if(dMag === 0) return {x:0, y:0};
      return {x: dirX/dMag, y: dirY/dMag};
    };

    const getShortestChord = (cx, cy) => {
      let bestP1 = null, bestP2 = null;
      let shortestDist = Infinity;
      let bestAngle = 0;
      for (let i = 0; i < trigCache.length; i++) {
        const { dx, dy, angle } = trigCache[i];

        const r1 = castRay(cx, cy, dx, dy, maxCut);
        const r2 = castRay(cx, cy, -dx, -dy, maxCut);

        if (r1.p && r2.p && r1.dist < maxCut && r2.dist < maxCut) {
          const totalDist = r1.dist + r2.dist;
          if (totalDist < shortestDist) {
            shortestDist = totalDist;
            bestP1 = r1.p; bestP2 = r2.p;
            bestAngle = angle;
          }
        }
      }
      return { dist: shortestDist, p1: bestP1, p2: bestP2, angle: bestAngle };
    };

    const candidates = [];
    const gridStep = Math.max(5, Math.floor(spacing / 3));

    for (let y = gridStep; y < h; y += gridStep) {
      for (let x = gridStep; x < w; x += gridStep) {
        if (!isWall(x, y)) {
          const chord = getShortestChord(x, y);
          if (chord.dist >= minCut && chord.dist <= maxCut) {
            const mx = (chord.p1.x + chord.p2.x) / 2;
            const my = (chord.p1.y + chord.p2.y) / 2;
            const perpRad = ((chord.angle + 90) % 360) * Math.PI / 180;
            const px = Math.cos(perpRad);
            const py = Math.sin(perpRad);
            
            const space1 = castRay(mx, my, px, py, maxCut).dist;
            const space2 = castRay(mx, my, -px, -py, maxCut).dist;
            const minDepth = Math.max(minCut * 1.5, chord.dist * 0.4);

            if (space1 > minDepth && space2 > minDepth) {
              candidates.push({ x, y, chord });
            }
          }
        }
      }
    }

    candidates.sort((a, b) => a.chord.dist - b.chord.dist);

    let drawnCount = 0;
    const drawnMidpoints = [];
    const newPaths = []; 

    const burnLineToMask = (x0, y0, x1, y1) => {
      const dX = x1 - x0; const dY = y1 - y0;
      const dist = Math.sqrt(dX * dX + dY * dY);
      const steps = Math.max(1, Math.ceil(dist * 2)); 
      const dx = dX / steps; const dy = dY / steps;
      let x = x0, y = y0;
      const radius = Math.max(1, Math.floor(thick / 2));

      for (let i = 0; i <= steps; i++) {
        const rx = (x + 0.5) | 0; // Ultra-fast bitwise round extracted from inner loop
        const ry = (y + 0.5) | 0;
        
        for (let oy = -radius; oy <= radius; oy++) {
          const py = ry + oy;
          if (py >= 0 && py < h) {
            const rowOffset = py * w; // Pre-calculate row multiplier
            for (let ox = -radius; ox <= radius; ox++) {
              const px = rx + ox;
              if (px >= 0 && px < w) {
                const idx = (rowOffset + px) << 2; // Fast bitwise shift instead of * 4
                if (data[idx+3] >= 128) {
                  data[idx] = 0; data[idx+1] = 0; data[idx+2] = 0; 
                }
              }
            }
          }
        }
        x += dx; y += dy;
      }
    };

    for (const cand of candidates) {
      if (isWall(cand.x, cand.y)) continue;

      // Pure Geometric Density Logic:
      // cand.chord.dist represents how "thick" or "wide" the current area is.
      const localWidth = cand.chord.dist;
      
      // If the area is narrow (like water), it uses your base 'spacing' slider setting.
      // If the area is massive (like the bag), we multiply the required spacing.
      // A 1.8x multiplier forces the algorithm to leave huge gaps in wide areas.
      const adaptiveSpacing = Math.max(spacing, localWidth * 1.8); 
      const spacingSq = adaptiveSpacing * adaptiveSpacing;

      let tooClose = false;
      for (const mid of drawnMidpoints) {
        const dx = cand.x - mid.x; const dy = cand.y - mid.y;
        if ((dx * dx + dy * dy) < spacingSq) {
          tooClose = true; break;
        }
      }
      if (tooClose) continue;

      const liveChord = getShortestChord(cand.x, cand.y);
      
      if (liveChord.dist >= minCut && liveChord.dist <= maxCut) {
        const mx = (liveChord.p1.x + liveChord.p2.x) / 2;
        const my = (liveChord.p1.y + liveChord.p2.y) / 2;
        const perpRad = ((liveChord.angle + 90) % 360) * Math.PI / 180;
        const px = Math.cos(perpRad);
        const py = Math.sin(perpRad);
        
        const space1 = castRay(mx, my, px, py, maxCut).dist;
        const space2 = castRay(mx, my, -px, -py, maxCut).dist;
        const minDepth = Math.max(minCut * 1.5, liveChord.dist * 0.4);

        if (space1 > minDepth && space2 > minDepth) {
          const chordDist = liveChord.dist;
          const n1 = getInwardNormal(liveChord.p1.x, liveChord.p1.y, cand.x, cand.y);
          const n2 = getInwardNormal(liveChord.p2.x, liveChord.p2.y, cand.x, cand.y);

          const cp1x = liveChord.p1.x + n1.x * (chordDist * smoothness);
          const cp1y = liveChord.p1.y + n1.y * (chordDist * smoothness);
          const cp2x = liveChord.p2.x + n2.x * (chordDist * smoothness);
          const cp2y = liveChord.p2.y + n2.y * (chordDist * smoothness);

          newPaths.push({
            id: Date.now() + Math.random(),
            type: 'auto',
            isEraser: false,
            color: color,
            thickness: thick,
            p1: { x: liveChord.p1.x, y: liveChord.p1.y },
            p2: { x: liveChord.p2.x, y: liveChord.p2.y },
            cp1: smoothness > 0.02 ? { x: cp1x, y: cp1y } : null,
            cp2: smoothness > 0.02 ? { x: cp2x, y: cp2y } : null,
          });

          if (smoothness > 0.02) {
            let prevX = liveChord.p1.x, prevY = liveChord.p1.y;
            const steps = Math.max(8, Math.floor(chordDist)); 
            for (let i = 1; i <= steps; i++) {
              const t = i / steps, mt = 1 - t;
              const cx = mt*mt*mt*liveChord.p1.x + 3*mt*mt*t*cp1x + 3*mt*t*t*cp2x + t*t*t*liveChord.p2.x;
              const cy = mt*mt*mt*liveChord.p1.y + 3*mt*mt*t*cp1y + 3*mt*t*t*cp2y + t*t*t*liveChord.p2.y;
              burnLineToMask(prevX, prevY, cx, cy);
              prevX = cx; prevY = cy;
            }
          } else {
            burnLineToMask(liveChord.p1.x, liveChord.p1.y, liveChord.p2.x, liveChord.p2.y);
          }
          drawnMidpoints.push({ x: mx, y: my });
          drawnCount++;
        }
      }
    }

    renderAllPaths(maskCtx, w, h, newPaths);
    
    // Execute Final Pipeline Render logic through our central helper
    imgObj.paths = newPaths; 
    const result = await regenerateBlobsForImage(imgObj, newPaths);
    
    // FORCE GARBAGE COLLECTION: Destroy processing canvases & source image
    canvas.width = 0; canvas.height = 0;
    maskCanvas.width = 0; maskCanvas.height = 0;
    img.src = '';
    
    return {
        ...result,
        paths: newPaths,
        history: [newPaths],
        historyIndex: 0,
        autoStats: { thick, lines: drawnCount, time: Math.round(performance.now() - startTime) }
    };
  };

      l = new Image(); l.src = img.resultLettersURL;
      await new Promise(r => l.onload = r);
      ctx.drawImage(l, 0, 0);
    }
    const rawBlob = await new Promise(res => c.toBlob(res, 'image/png'));
    
    // Memory Cleanup
    c.width = 0; c.height = 0;
    if (m) m.src = '';
    if (l) l.src = '';
    
    return injectPngDpi(rawBlob, img.dpi || 300); // Preserve high DPI
  };

  const downloadCombinedSingle = async (img) => {
    setIsProcessing(true);
    setDropdownOpen(null);
    try {
      const blob = await getCombinedBlob(img);
      const url = URL.createObjectURL(blob);
      handleDownload(url, 'Combined', img.name);
      setTimeout(() => URL.revokeObjectURL(url), 2000); // cleanup
    } catch (err) { console.error(err); }
    setIsProcessing(false);
  };

  const loadJSZip = async () => {
    if (window.JSZip) return window.JSZip;
    return new Promise((resolve, reject) => {
      const script = document.createElement('script');
      script.src = 'https://cdnjs.cloudflare.com/ajax/libs/jszip/3.10.1/jszip.min.js';
      script.onload = () => resolve(window.JSZip);
      script.onerror = () => {
        console.error("Failed to load JSZip");
        reject(new Error("Failed to load JSZip"));
      };
      document.head.appendChild(script);
    });
  };

  const handleDownloadZip = async (type) => {
    setIsProcessing(true);
    setDropdownOpen(null);
    try {
      const JSZip = await loadJSZip();
      const zip = new JSZip();
      
      const typeLabelMap = {
        'art': 'FullArt',
        'base': 'BaseArt',
        'mask': 'LinesOnly',
        'letters': 'LettersOnly',
        'combined': 'Combined'
      };
      const typeLabel = typeLabelMap[type] || 'Export';
      
      const targetImages = imagesRef.current.filter(img => 
        type === 'mask' ? img.resultMaskURL : 
        type === 'letters' ? img.resultLettersURL : 
        type === 'combined' ? (img.resultMaskURL || img.resultLettersURL) : 
        type === 'base' ? img.dataUrl : 
        img.resultImage
      );
      
      for (let i = 0; i < targetImages.length; i++) {
        const img = targetImages[i];
        let blob;
        
        if (type === 'combined') {
           blob = await getCombinedBlob(img);
        } else {
           const url = type === 'mask' ? img.resultMaskURL : type === 'letters' ? img.resultLettersURL : type === 'base' ? (img.resultBaseURL || img.dataUrl) : img.resultImage;
           blob = await fetch(url).then(r => r.blob());
        }
        
        const formattedName = formatExportName(img.name);
        zip.file(`${i + 1}_${formattedName}_MonoSegment_${typeLabel}.png`, blob);
      }
      
      const zipBlob = await zip.generateAsync({ type: 'blob' });
      const zipUrl = URL.createObjectURL(zipBlob);
      
      const now = new Date();
      const dateStr = `${now.getFullYear()}-${String(now.getMonth() + 1).padStart(2, '0')}-${String(now.getDate()).padStart(2, '0')}_${String(now.getHours()).padStart(2, '0')}-${String(now.getMinutes()).padStart(2, '0')}`;
      
      const a = document.createElement('a');
      a.href = zipUrl;
      a.download = `MonoSegment_${typeLabel}_Batch_${dateStr}.zip`;
      document.body.appendChild(a);
      a.click();
      document.body.removeChild(a);
      URL.revokeObjectURL(zipUrl);
    } catch (err) {
      console.error("ZIP Generation Failed:", err);
    } finally {
      setIsProcessing(false);
    }
  };

  // --- 🖌 MANUAL EDITING LOGIC ---
  const startEditing = () => {
    // Take a pristine snapshot of the vectors before the user touches anything
    if (activeImg) {
       preEditSnapshotRef.current = { 
           paths: JSON.parse(JSON.stringify(activeImg.paths)), 
           history: JSON.parse(JSON.stringify(activeImg.history)), 
           historyIndex: activeImg.historyIndex,
           letters: activeImg.letters ? JSON.parse(JSON.stringify(activeImg.letters)) : [], // BUG FIX: Backup letters
           hasCustomLetters: activeImg.hasCustomLetters
       };
       
       // FIX: Sync the memory tracker so it knows about the auto-generated lines!
       historyRef.current = activeImg.history || [[]];
       historyIndexRef.current = activeImg.historyIndex || 0;
       updateHistoryState();
    }
    setViewerMode('edit');
    setPenNodes([]);
    setPreviewPos(null);
    isDraggingHandle.current = false;
    setSelectedLetterId(null); // FIX: Clear ghost selections
  };

  const saveEdits = async () => {
    setIsProcessing(true);
    try {
      if (penNodes.length > 0) commitPenPath(); 
      const currentPaths = historyRef.current[historyIndexRef.current];
      const newBlobs = await regenerateBlobsForImage(activeImg, currentPaths);
      
      updateActiveImage({ ...newBlobs, paths: currentPaths });
      setViewerMode('preview');
      preEditSnapshotRef.current = null; // Clean up memory
      setSelectedLetterId(null); // FIX: Clear ghost selections
    } catch (e) {
      console.error("Failed saving edits", e);
    } finally {
      setIsProcessing(false);
    }
  };

  const cancelEdits = () => {
    // Nuke the corrupted state and instantly restore the pristine snapshot
    if (preEditSnapshotRef.current) {
        updateActiveImage({ 
            paths: preEditSnapshotRef.current.paths, 
            history: preEditSnapshotRef.current.history, 
            historyIndex: preEditSnapshotRef.current.historyIndex,
            letters: preEditSnapshotRef.current.letters, // BUG FIX: Restore letters
            hasCustomLetters: preEditSnapshotRef.current.hasCustomLetters
        });
        historyRef.current = preEditSnapshotRef.current.history;
        historyIndexRef.current = preEditSnapshotRef.current.historyIndex;
        updateHistoryState();
    }
    
    // Force-kill any active drawing states to prevent mouse-up race conditions
    setIsDrawing(false);
    drawPoints.current = [];
    isMagicErasing.current = false;
    
    setViewerMode('preview');
    setPenNodes([]);
    isDraggingHandle.current = false;
    preEditSnapshotRef.current = null; // Clear snapshot
    setSelectedLetterId(null); // FIX: Clear ghost selections
  };

  const getCanvasCoordinates = (e) => {
    if (!overlayCanvasRef.current) return { x: 0, y: 0 };
    const canvas = overlayCanvasRef.current;
    const rect = canvas.getBoundingClientRect();
    const clientX = e.touches ? e.touches[0].clientX : e.clientX;
    const clientY = e.touches ? e.touches[0].clientY : e.clientY;
    // Calculate physical canvas coordinate, then mathematically subtract the margin so the database remains pure
    return { 
       x: (clientX - rect.left) / zoom - OVERFLOW_MARGIN, 
       y: (clientY - rect.top) / zoom - OVERFLOW_MARGIN 
    };
  };

  const commitPenPath = () => {
    if (penNodes.length < 2) {
      setPenNodes([]);
      setPreviewPos(null);
      isDraggingHandle.current = false;
      return;
    }
    
    const newPath = {
      id: crypto.randomUUID(),
      type: 'pen',
      isEraser: false,
      color: lineColor,
      thickness: Number(lineThickness) || 1,
      nodes: [...penNodes]
    };

    updatePathsAndHistory([...activeImg.paths, newPath]);
    setPenNodes([]);
    setPreviewPos(null);
    isDraggingHandle.current = false;
  };

  // Live render overlay
  useEffect(() => {
    if (viewerMode !== 'edit' || !overlayCanvasRef.current) return;
    const ctx = overlayCanvasRef.current.getContext('2d');
    const w = overlayCanvasRef.current.width;
    const h = overlayCanvasRef.current.height;
    ctx.clearRect(0, 0, w, h);

    ctx.save();
    ctx.translate(OVERFLOW_MARGIN, OVERFLOW_MARGIN); // Shift the live view by the margin

    const thick = Number(lineThickness) || 1;
    const eSize = Number(eraserSize) || 9;

    if ((editTool === 'brush' || editTool === 'gapBrush' || (editTool === 'erase' && eraserMode === 'brush')) && previewPos && !isSpacePressed) {
      const radius = (editTool === 'erase' || editTool === 'gapBrush' ? eSize : thick) / 2;
      
      ctx.beginPath();
      ctx.arc(previewPos.x, previewPos.y, radius + (1 / zoom), 0, Math.PI * 2);
      ctx.lineWidth = 1.5 / zoom; 
      ctx.strokeStyle = 'rgba(255, 255, 255, 0.8)';
      ctx.stroke();

      ctx.beginPath();
      ctx.arc(previewPos.x, previewPos.y, radius, 0, Math.PI * 2);
      ctx.lineWidth = 1.5 / zoom;
      ctx.strokeStyle = (editTool === 'erase' || editTool === 'gapBrush') ? '#000000' : lineColor;
      ctx.stroke();
    }

    if (editTool === 'pen' && penNodes.length > 0) {
      ctx.beginPath();
      ctx.strokeStyle = lineColor;
      ctx.lineWidth = thick;
      ctx.lineCap = 'round';
      ctx.lineJoin = 'round';
      
      ctx.moveTo(penNodes[0].x, penNodes[0].y);
      for (let i = 1; i < penNodes.length; i++) {
        const prev = penNodes[i - 1];
        const curr = penNodes[i];
        ctx.bezierCurveTo(prev.handleOut.x, prev.handleOut.y, curr.handleIn.x, curr.handleIn.y, curr.x, curr.y);
      }

      if (previewPos && !isDraggingHandle.current) {
        const prev = penNodes[penNodes.length - 1];
        ctx.bezierCurveTo(prev.handleOut.x, prev.handleOut.y, previewPos.x, previewPos.y, previewPos.x, previewPos.y);
      }
      ctx.stroke();

      const handleScale = 1 / zoom; 
      
      penNodes.forEach((p, idx) => {
        const hasPulledHandle = p.handleOut.x !== p.x || p.handleOut.y !== p.y;
        if (hasPulledHandle) {
          ctx.beginPath();
          ctx.moveTo(p.handleIn.x, p.handleIn.y);
          ctx.lineTo(p.handleOut.x, p.handleOut.y);
          ctx.strokeStyle = '#a5b4fc';
          ctx.lineWidth = 1 * handleScale;
          ctx.stroke();

          ctx.beginPath();
          ctx.arc(p.handleIn.x, p.handleIn.y, 3 * handleScale, 0, Math.PI * 2);
          ctx.arc(p.handleOut.x, p.handleOut.y, 3 * handleScale, 0, Math.PI * 2);
          ctx.fillStyle = '#ffffff';
          ctx.strokeStyle = '#6366f1';
          ctx.fill();
          ctx.stroke();
        }

        ctx.beginPath();
        ctx.rect(p.x - (3 * handleScale), p.y - (3 * handleScale), 6 * handleScale, 6 * handleScale);
        ctx.fillStyle = idx === penNodes.length - 1 ? '#6366f1' : '#ffffff';
        ctx.strokeStyle = '#6366f1';
        ctx.lineWidth = 1.5 * handleScale;
        ctx.fill();
        ctx.stroke();
      });
    }

    // NEW: Draw a dashed bounding box around the selected letter so the user knows it's active
    if (editTool === 'moveLetter' && selectedLetterId && activeImg && activeImg.letters) {
      const selectedLetter = activeImg.letters.find(l => l.id === selectedLetterId);
      if (selectedLetter) {
        ctx.font = `bold ${selectedLetter.size}px ${selectedLetter.font}`;
        const metrics = ctx.measureText(selectedLetter.char);
        const lw = metrics.width;
        const lh = selectedLetter.size;
        
        ctx.strokeStyle = '#6366f1'; // Indigo 500
        ctx.lineWidth = 2 / zoom;
        ctx.setLineDash([4 / zoom, 4 / zoom]);
        ctx.strokeRect(selectedLetter.x - lw / 2 - 4, selectedLetter.y - lh / 2 - 4, lw + 8, lh + 8);
        ctx.setLineDash([]); // Reset dash
      }
    }
    
    ctx.restore(); // Clean up the shift so we don't infinitely stack it
  }, [penNodes, previewPos, editTool, viewerMode, lineColor, lineThickness, zoom, eraserSize, eraserMode, isSpacePressed, selectedLetterId, activeImg?.letters]);

  const handleMagicEraseVector = (x, y) => {
    if (!activeImg) return;
    const ctx = editCanvasRef.current.getContext('2d');
    
    let hitIndex = -1;
    const currentPaths = activeImg.paths || [];

    for (let i = currentPaths.length - 1; i >= 0; i--) {
      const p = currentPaths[i];
      if (p.isEraser) continue; 

      const path2d = buildPath2D(p);
      ctx.lineWidth = p.thickness + 24; 
      if (ctx.isPointInStroke(path2d, x, y)) {
          hitIndex = i;
          break;
      }
    }

    if (hitIndex !== -1) {
      const newPaths = [...currentPaths];
      newPaths.splice(hitIndex, 1);
      updatePathsAndHistory(newPaths);
    }
  };

  const handleEditMouseDown = (e) => {
    if (isSpacePressed || e.button === 1 || e.button === 2) return; 
    e.stopPropagation();

    const { x, y } = getCanvasCoordinates(e);
    const thick = Number(lineThickness) || 1;
    const eSize = Number(eraserSize) || 9;
    
    if (editTool === 'moveLetter') {
      if (!activeImg || !activeImg.letters) return;
      const ctx = editCanvasRef.current.getContext('2d');
      const clickX = x + OVERFLOW_MARGIN;
      const clickY = y + OVERFLOW_MARGIN;
      
      let clickedId = null;
      // Reverse loop to select top-most letter if they overlap
      for (let i = activeImg.letters.length - 1; i >= 0; i--) {
         const l = activeImg.letters[i];
         ctx.font = `bold ${l.size}px ${l.font}`;
         const metrics = ctx.measureText(l.char);
         const w = metrics.width;
         const h = l.size; 
         // Simple bounding box hit detection
         if (Math.abs(clickX - (l.x + OVERFLOW_MARGIN)) <= w/2 && Math.abs(clickY - (l.y + OVERFLOW_MARGIN)) <= h/2) {
             clickedId = l.id;
             dragOffset.current = { dx: clickX - (l.x + OVERFLOW_MARGIN), dy: clickY - (l.y + OVERFLOW_MARGIN) };
             break;
         }
      }
      setSelectedLetterId(clickedId);
      if (clickedId) isDraggingLetter.current = true;
      return;
    } else if (editTool === 'pen') {
      isDraggingHandle.current = true;
      setPenNodes(prev => [...prev, { x, y, handleIn: { x, y }, handleOut: { x, y } }]);
    } else if (editTool === 'erase' && eraserMode === 'magic') {
      isMagicErasing.current = true;
      handleMagicEraseVector(x, y); 
    } else {
      setIsDrawing(true);
      drawPoints.current = [{ x, y }];
      lastDrawnPos.current = { x, y };

      const ctx = editCanvasRef.current.getContext('2d');
      ctx.lineCap = 'round';
      ctx.lineJoin = 'round';
      
      if (editTool === 'erase') {
        ctx.globalCompositeOperation = 'destination-out';
        ctx.strokeStyle = 'rgba(0,0,0,1)'; 
        ctx.fillStyle = 'rgba(0,0,0,1)';
        ctx.lineWidth = eSize;
      } else if (editTool === 'gapBrush') {
        ctx.globalCompositeOperation = 'source-over';
        ctx.strokeStyle = '#000000';
        ctx.fillStyle = '#000000';
        ctx.lineWidth = eSize;
      } else {
        ctx.globalCompositeOperation = 'source-over';
        ctx.strokeStyle = lineColor;
        ctx.fillStyle = lineColor;
        ctx.lineWidth = thick;
      }
      
      const radius = (editTool === 'erase' || editTool === 'gapBrush' ? eSize : thick) / 2;
      ctx.beginPath();
      ctx.arc(x + OVERFLOW_MARGIN, y + OVERFLOW_MARGIN, radius, 0, Math.PI * 2);
      ctx.fill();
    }
  };

  const handleEditMouseMove = (e) => {
    if (isSpacePressed) {
      setPreviewPos(null); 
      return;
    }
    const { x, y } = getCanvasCoordinates(e);

    if (editTool === 'moveLetter') {
       if (isDraggingLetter.current && selectedLetterId && activeImg) {
           const newLetters = activeImg.letters.map(l => {
               if (l.id === selectedLetterId) {
                   return { ...l, x: x - dragOffset.current.dx, y: y - dragOffset.current.dy, isManual: true }; // TAG AS MANUAL
               }
               return l;
           });
           updateActiveImage({ letters: newLetters, hasCustomLetters: true });
       }
       return;
    } else if (editTool === 'erase' && eraserMode === 'magic') {
      if (isMagicErasing.current) {
        handleMagicEraseVector(x, y);
      }
    } else if (editTool === 'pen') {
      if (isDraggingHandle.current) {
        setPenNodes(prev => {
          const newNodes = [...prev];
          const lastNode = newNodes[newNodes.length - 1];
          lastNode.handleOut = { x, y };
          lastNode.handleIn = { x: lastNode.x * 2 - x, y: lastNode.y * 2 - y };
          return newNodes;
        });
      } else {
        setPreviewPos({ x, y });
      }
    } else if (editTool === 'brush' || editTool === 'gapBrush' || (editTool === 'erase' && eraserMode === 'brush')) {
      setPreviewPos({ x, y }); 
      
      if (isDrawing) {
        const lastPt = drawPoints.current[drawPoints.current.length - 1];
        const dx = x - lastPt.x; const dy = y - lastPt.y;
        const dist = Math.sqrt(dx * dx + dy * dy);
        const skipThreshold = (freehandSmoothness / 100) * 20;

        if (dist > skipThreshold || freehandSmoothness === 0) {
           drawPoints.current.push({ x, y });
           const pts = drawPoints.current;
           
           if (pts.length >= 3 && lastDrawnPos.current) {
             const ctx = editCanvasRef.current.getContext('2d');
             const p2 = pts[pts.length - 2];
             const p3 = pts[pts.length - 1];
             const mid = { x: (p2.x + p3.x) / 2, y: (p2.y + p3.y) / 2 };

             ctx.beginPath();
             ctx.moveTo(lastDrawnPos.current.x + OVERFLOW_MARGIN, lastDrawnPos.current.y + OVERFLOW_MARGIN);
             ctx.quadraticCurveTo(p2.x + OVERFLOW_MARGIN, p2.y + OVERFLOW_MARGIN, mid.x + OVERFLOW_MARGIN, mid.y + OVERFLOW_MARGIN);
             ctx.stroke();

             lastDrawnPos.current = mid;
           }
        }
      }
    }
  };

  const handleEditMouseUp = (e) => {
    isMagicErasing.current = false;
    isDraggingLetter.current = false;
    
    if (editTool === 'pen') {
      if (isDraggingHandle.current) {
        isDraggingHandle.current = false;
        const { x, y } = getCanvasCoordinates(e);
        setPreviewPos({ x, y }); 
      }
      return;
    }

    if (!isDrawing) return;
    setIsDrawing(false);
    
    if (previewPos) {
       drawPoints.current.push(previewPos);
    }

    const pts = [...drawPoints.current];
    if (pts.length > 0 && activeImg) {
       const newPath = {
          id: crypto.randomUUID(),
          type: 'brush',
          isEraser: editTool === 'erase',
          isGapFill: editTool === 'gapBrush',
          color: editTool === 'erase' ? null : (editTool === 'gapBrush' ? '#000000' : lineColor),
          thickness: (editTool === 'erase' || editTool === 'gapBrush') ? (Number(eraserSize) || 9) : (Number(lineThickness) || 1),
          smoothness: freehandSmoothness,
          points: pts
       };
       updatePathsAndHistory([...activeImg.paths, newPath]);
    }
    
    drawPoints.current = [];
    lastDrawnPos.current = null;
  };

  const handleEditMouseLeave = (e) => {
    setPreviewPos(null); 
    handleEditMouseUp(e);
  };

  const handleEditContextMenu = (e) => {
    e.preventDefault();
    if (editTool === 'pen' && penNodes.length > 0) commitPenPath();
  };

  const PRESET_COLORS = [
    { label: 'Black', value: '#000000' },
    { label: 'Red', value: '#FF0000' },
    { label: 'Blue', value: '#0000FF' },
    { label: 'Green', value: '#00FF00' },
    { label: 'White', value: '#FFFFFF' },
    { label: 'Gray', value: '#808080' },
  ];

  return (
    <div className="h-screen flex overflow-hidden bg-gray-100 select-none font-sans">
      
      {/* --- KEYBOARD SHORTCUTS MODAL --- */}
      {showShortcuts && (
        <div className="fixed inset-0 bg-black/60 z-[100] flex items-center justify-center backdrop-blur-sm animate-in fade-in duration-200">
          <div className="bg-white rounded-2xl w-[500px] overflow-hidden shadow-2xl flex flex-col scale-in-center">
            <div className="px-6 py-4 border-b border-neutral-200 flex items-center justify-between bg-neutral-50">
              <h2 className="text-lg font-bold text-neutral-800 flex items-center gap-2"><Keyboard className="w-5 h-5 text-indigo-600"/> Keyboard Shortcuts</h2>
              <button onClick={() => setShowShortcuts(false)} className="text-neutral-400 hover:text-neutral-800 transition-colors p-1 hover:bg-neutral-200 rounded"><X className="w-5 h-5"/></button>
            </div>
            <div className="p-6 space-y-4">
              <div className="grid grid-cols-2 gap-4">
                <div className="space-y-3">
                  <h3 className="text-xs font-bold text-neutral-400 uppercase tracking-wider mb-2">Workspace</h3>
                  <div className="flex items-center justify-between text-sm"><span className="text-neutral-600">Pan Canvas</span><kbd className="bg-neutral-100 border border-neutral-300 text-neutral-600 px-2 py-1 rounded text-xs font-mono">Space + Drag</kbd></div>
                  <div className="flex items-center justify-between text-sm"><span className="text-neutral-600">Zoom In/Out</span><kbd className="bg-neutral-100 border border-neutral-300 text-neutral-600 px-2 py-1 rounded text-xs font-mono">Mouse Wheel</kbd></div>
                  <div className="flex items-center justify-between text-sm"><span className="text-neutral-600">Undo Action</span><kbd className="bg-neutral-100 border border-neutral-300 text-neutral-600 px-2 py-1 rounded text-xs font-mono">Ctrl + Z</kbd></div>
                </div>
                <div className="space-y-3">
                  <h3 className="text-xs font-bold text-neutral-400 uppercase tracking-wider mb-2">Tools</h3>
                  <div className="flex items-center justify-between text-sm"><span className="text-neutral-600">Refine Tool</span><kbd className="bg-neutral-100 border border-neutral-300 text-neutral-600 px-2 py-1 rounded text-xs font-mono">R</kbd></div>
                  <div className="flex items-center justify-between text-sm"><span className="text-neutral-600">Pen Tool</span><kbd className="bg-neutral-100 border border-neutral-300 text-neutral-600 px-2 py-1 rounded text-xs font-mono">P</kbd></div>
                  <div className="flex items-center justify-between text-sm"><span className="text-neutral-600">Eraser</span><kbd className="bg-neutral-100 border border-neutral-300 text-neutral-600 px-2 py-1 rounded text-xs font-mono">E</kbd></div>
                </div>
              </div>
            </div>
            <div className="px-6 py-4 bg-neutral-50 border-t border-neutral-200 flex justify-end">
              <button onClick={() => setShowShortcuts(false)} className="px-4 py-2 bg-neutral-800 hover:bg-neutral-900 text-white rounded-lg text-sm font-bold transition-colors">Got it</button>
            </div>
          </div>
        </div>
      )}

      {/* --- STARTUP MODAL OVERLAY --- */}
      {showStartupModal && (
        <div className="fixed inset-0 z-[100] bg-black/60 backdrop-blur-sm flex items-center justify-center p-4">
          <div className="bg-white rounded-2xl shadow-2xl max-w-md w-full p-6 text-center animate-in fade-in zoom-in duration-200">
            <div className="w-16 h-16 bg-indigo-100 rounded-full flex items-center justify-center mx-auto mb-4 text-indigo-600">
              <HardDrive className="w-8 h-8" />
            </div>
            <h2 className="text-2xl font-bold text-neutral-800 mb-2">Project Found</h2>
            <p className="text-sm text-neutral-600 mb-6">
              We found a previously saved MonoSegment project. Would you like to load it where you left off, or start a brand new project?
            </p>
            <div className="flex flex-col gap-3">
              <button 
                onClick={loadProject}
                className="w-full py-3 px-4 bg-indigo-600 hover:bg-indigo-700 text-white rounded-lg font-bold flex items-center justify-center gap-2 transition-all shadow-md active:scale-[0.98]"
              >
                <Cloud className="w-5 h-5" /> Load Saved Project
              </button>
              <button 
                onClick={startFresh}
                className="w-full py-3 px-4 bg-neutral-100 hover:bg-neutral-200 text-neutral-700 rounded-lg font-bold flex items-center justify-center gap-2 transition-all active:scale-[0.98]"
              >
                <X className="w-5 h-5" /> Start Fresh (Discard)
              </button>
            </div>
          </div>
        </div>
      )}

      {/* --- CLEAR WORKSPACE MODAL --- */}
      {showClearConfirm && (
        <div className="fixed inset-0 z-[100] bg-black/60 backdrop-blur-sm flex items-center justify-center p-4">
          <div className="bg-white rounded-2xl shadow-2xl max-w-md w-full p-6 text-center animate-in fade-in zoom-in duration-200">
            <div className="w-16 h-16 bg-red-100 rounded-full flex items-center justify-center mx-auto mb-4 text-red-600">
              <Trash2 className="w-8 h-8" />
            </div>
            <h2 className="text-2xl font-bold text-neutral-800 mb-2">Clear Workspace?</h2>
            <p className="text-sm text-neutral-600 mb-6">
              Are you sure you want to remove all images from the workspace? Unsaved changes to these files will be lost.

              {(editTool === 'brush' || editTool === 'gapBrush' || (editTool === 'erase' && eraserMode === 'brush')) && (
                <div className="flex items-center gap-2 bg-white px-2 py-1.5 rounded-md border border-neutral-200 shadow-sm ml-1 min-w-[120px]">
                  <span className="text-xs font-medium text-neutral-500">Smooth:</span>
                  <input 
                    type="range" min="0" max="100" step="1" value={freehandSmoothness} 
                    onChange={e => setFreehandSmoothness(Number(e.target.value))} 
                    className="w-16 accent-purple-600 flex-1"
                    title="Higher values stabilize jagged mouse movements"
                  />
                </div>
              )}
              
              <div className="flex items-center gap-2 bg-white px-2 py-1.5 rounded-md border border-neutral-200 shadow-sm ml-1 min-w-[100px]">
                <span className="text-xs font-medium text-neutral-500">Size:</span>
                {(editTool === 'erase' && eraserMode === 'brush') || editTool === 'gapBrush' ? (
                  <input 
                    type="range" min="1" max="50" step="0.5" value={eraserSize} 
                    onChange={e => setEraserSize(Number(e.target.value))} 
                    className="w-16 accent-purple-600 flex-1"
                  />
                ) : editTool === 'erase' && eraserMode === 'magic' ? (
                  <span className="text-xs font-bold text-neutral-400 flex-1 text-center">Auto</span>
                ) : (
                  <span className="text-xs font-bold text-neutral-700 flex-1 text-center">{lineThickness}px</span>
                )}
              </div>

              <div className="flex gap-2 ml-2 border-l border-purple-200 pl-3">
                <button 
                  onClick={cancelEdits} 
                  className="text-xs flex items-center gap-1 px-3 py-1.5 bg-white border border-neutral-300 rounded-md hover:bg-neutral-50 text-neutral-600 font-medium transition-colors"
                >
                  <X className="w-3.5 h-3.5" /> Cancel
                </button>
                <button 
                  id="apply-btn"
                  onClick={saveEdits} 
                  disabled={isProcessing}
                  className="text-xs flex items-center gap-1 px-3 py-1.5 bg-purple-600 border border-purple-700 rounded-md hover:bg-purple-700 text-white font-medium transition-colors shadow-sm disabled:opacity-50"
                  title="Save and Apply (Ctrl+Enter)"
                >
                  {isProcessing ? <Loader2 className="w-3.5 h-3.5 animate-spin" /> : <Check className="w-3.5 h-3.5" />}
                  Apply
                </button>
              </div>
            </div>
          </div>
        )}

        {/* Workspace Physical Artboard Area */}
        <div 
          id="preview-workspace" 
          className={`flex-1 overflow-auto relative outline-none ${isPanning ? 'cursor-grabbing' : isSpacePressed ? 'cursor-grab' : ''}`}
          ref={workspaceRef}
          onMouseDown={handleWorkspaceMouseDown}
        >
          {/* Inner container scales up with zoom */}
          <div 
            className="relative transition-opacity duration-200"
            style={{ 
               width: activeImg ? Math.max(workspaceSize.w, imageSize.w * zoom + 60) : '100%', 
               height: activeImg ? Math.max(workspaceSize.h, imageSize.h * zoom + 60) : '100%',
               opacity: isProcessing ? 0.6 : 1 
            }}
          >
            
            {/* Physical bounding box of the art */}
            <div 
              id="physical-page"
              className={`absolute origin-center ${activeImg ? 'shadow-2xl' : ''}`}
              onContextMenu={viewerMode === 'edit' ? handleEditContextMenu : undefined}
              style={{
                left: '50%',
                top: '50%',
                transform: 'translate(-50%, -50%)',
                width: activeImg ? (imageSize.w * zoom) : '100%',
                height: activeImg ? (imageSize.h * zoom) : '100%',
                backgroundColor: activeImg ? (viewerBg === 'transparent' ? '#E4E6EA' : viewerBg) : 'transparent',
                backgroundImage: activeImg && viewerBg === 'transparent' ? 'radial-gradient(#b4b4b4 1px, transparent 1px)' : 'none', 
                backgroundSize: `${20 * zoom}px ${20 * zoom}px`
              }}
            >
              
              {!activeImg && (
                <div className="absolute inset-0 flex flex-col items-center justify-center text-neutral-400">
                  <ImageIcon className="w-12 h-12 opacity-30 mb-2" />
                  <p className="text-xs font-semibold">Workspace (Upload to begin)</p>
                </div>
              )}

              {/* Preview Mode Render */}
              {activeImg && viewerMode === 'preview' && (
                <img 
                  src={showProcessed && activeImg.resultImage ? activeImg.resultImage : activeImg.dataUrl} 
                  alt="Artboard" 
                  className="w-full h-full object-fill pointer-events-none" 
                />
              )}

              {/* Edit Mode Render */}
              {viewerMode === 'edit' && activeImg && (
                <>
                  <img 
                    src={activeImg.dataUrl} 
                    alt="Base" 
                    className="w-full h-full object-fill pointer-events-none" 
                  />
                  
                  {/* Commits applied here */}
                  <canvas
                    ref={editCanvasRef}
                    width={imageSize.w + (OVERFLOW_MARGIN * 2)}
                    height={imageSize.h + (OVERFLOW_MARGIN * 2)}
                    className="absolute object-fill pointer-events-none"
                    style={{ 
                        left: -OVERFLOW_MARGIN * zoom,
                        top: -OVERFLOW_MARGIN * zoom,
                        width: (imageSize.w + (OVERFLOW_MARGIN * 2)) * zoom, 
                        height: (imageSize.h + (OVERFLOW_MARGIN * 2)) * zoom, 
                        mixBlendMode: lineBlendMode === 'multiply' ? 'multiply' : 'normal' 
                    }}
                  />
                  
                  {/* Interactive overlay */}
                  <canvas
                    ref={overlayCanvasRef}
                    width={imageSize.w + (OVERFLOW_MARGIN * 2)}
                    height={imageSize.h + (OVERFLOW_MARGIN * 2)}
                    className={`absolute object-fill touch-none ${
                      isSpacePressed ? '' :
                      (editTool === 'brush' || editTool === 'gapBrush' || (editTool === 'erase' && eraserMode === 'brush')) ? 'cursor-none' : 
                      'cursor-crosshair'
                    }`}
                    style={{ 
                        left: -OVERFLOW_MARGIN * zoom,
                        top: -OVERFLOW_MARGIN * zoom,
                        width: (imageSize.w + (OVERFLOW_MARGIN * 2)) * zoom, 
                        height: (imageSize.h + (OVERFLOW_MARGIN * 2)) * zoom, 
                        mixBlendMode: lineBlendMode === 'multiply' ? 'multiply' : 'normal' 
                    }}
                    onMouseDown={handleEditMouseDown}
                    onMouseMove={handleEditMouseMove}
                    onMouseUp={handleEditMouseUp}
                    onMouseLeave={handleEditMouseLeave}
                    onTouchStart={handleEditMouseDown}
                    onTouchMove={handleEditMouseMove}
                    onTouchEnd={handleEditMouseUp}
                  />
                </>
              )}

            </div>
          </div>

          </div>

        {/* Floating Letter Size Control Toast */}
        {viewerMode === 'edit' && editTool === 'moveLetter' && selectedLetterId && activeImg?.letters && (
           <div className="absolute bottom-32 left-1/2 -translate-x-1/2 bg-neutral-900 text-white px-4 py-2 rounded-xl shadow-2xl flex items-center gap-3 animate-in slide-in-from-bottom-4 fade-in z-[100] border border-neutral-700">
             <Type className="w-4 h-4 text-indigo-400" />
             <span className="text-xs font-bold text-neutral-300">Letter Size</span>
             <div className="flex items-center gap-1 bg-neutral-800 rounded-lg border border-neutral-700 p-0.5">
                 <button 
                     onClick={() => {
                         updateActiveImage({ letters: activeImg.letters.map(x => x.id === selectedLetterId ? { ...x, size: Math.max(1, x.size - 1), isManual: true } : x), hasCustomLetters: true });
                     }}
                     className="p-1.5 hover:bg-neutral-700 rounded text-neutral-400 hover:text-white transition-colors font-mono font-bold text-sm leading-none flex items-center justify-center w-7 h-7"
                     title="Decrease Size ( [ or - )"
                 >-</button>
                 
                 <div className="flex items-center bg-transparent">
                     <input 
                         type="number" 
                         value={activeImg.letters.find(x => x.id === selectedLetterId)?.size ?? ''}
                         onChange={(e) => {
                             let val = parseInt(e.target.value, 10);
                             if (isNaN(val)) val = ''; // Allow empty string so user can clear the box
                             if (val !== '' && val > 3000) val = 3000; // Cap maximum size to prevent browser crashing
                             
                             updateActiveImage({ letters: activeImg.letters.map(x => x.id === selectedLetterId ? { ...x, size: val, isManual: true } : x), hasCustomLetters: true });
                         }}
                         onBlur={(e) => {
                             // If they click away while the box is empty, snap it back to a safe minimum
                             if (e.target.value === '' || parseInt(e.target.value, 10) < 1) {
                                 updateActiveImage({ letters: activeImg.letters.map(x => x.id === selectedLetterId ? { ...x, size: 10, isManual: true } : x), hasCustomLetters: true });
                             }
                         }}
                         className="w-10 bg-transparent text-right text-sm font-bold focus:outline-none appearance-none p-0"
                     />
                     <span className="text-xs text-neutral-500 font-medium ml-0.5 mr-2 select-none">px</span>
                 </div>
                 
                 <button 
                     onClick={() => {
                         updateActiveImage({ letters: activeImg.letters.map(x => x.id === selectedLetterId ? { ...x, size: x.size + 1, isManual: true } : x), hasCustomLetters: true });
                     }}
                     className="p-1.5 hover:bg-neutral-700 rounded text-neutral-400 hover:text-white transition-colors font-mono font-bold text-sm leading-none flex items-center justify-center w-7 h-7"
                     title="Increase Size ( ] or + )"
                 >+</button>
             </div>
          </div>
        )}

        {/* Filmstrip / Multi-Image Dock */}
        {images.length > 0 && (
          <div className={`h-24 bg-white border-t border-gray-300 flex items-center px-4 gap-3 overflow-x-auto flex-shrink-0 z-20 shadow-[0_-4px_10px_rgba(0,0,0,0.02)] scrollbar-hide transition-opacity ${viewerMode === 'edit' ? 'opacity-50 pointer-events-none' : ''}`}>
            {images.map((img, i) => (
               <div 
                 key={img.id} 
                 id={`thumb-${i}`}
                 className={`group relative bg-neutral-900 w-16 h-16 rounded-md border-2 overflow-hidden flex-shrink-0 cursor-pointer transition-all ${i === activeIndex ? 'border-indigo-600 shadow-md scale-105' : 'border-gray-200 hover:border-indigo-400'}`}
                 onClick={() => setActiveIndex(i)}
                 title={img.name}
               >
                  <img src={img.dataUrl} className="w-full h-full object-contain" alt="thumbnail" />
                  
                  {img.resultImage && (
                     <div className="absolute bottom-1.5 left-1.5 bg-emerald-500 rounded-full w-2.5 h-2.5 border-2 border-white shadow-sm" title="Processed"></div>
                  )}
                  <button 
                    onClick={(e) => { e.stopPropagation(); deleteImage(i); }}
                    className="absolute top-1 right-1 bg-red-500 text-white p-0.5 rounded-full opacity-0 group-hover:opacity-100 transition-all hover:bg-red-600 hover:scale-110 shadow-sm flex items-center justify-center"
                    title="Remove Image"
                  >
                    <X className="w-3 h-3" />
                  </button>
               </div>
            ))}
            
            <label className="w-16 h-16 rounded-md border-2 border-dashed border-gray-300 flex items-center justify-center text-gray-400 hover:text-indigo-600 hover:border-indigo-400 hover:bg-indigo-50 cursor-pointer flex-shrink-0 transition-colors ml-1" title="Add more images">
               <input type="file" multiple accept="image/png, image/jpeg, image/webp, image/svg+xml" className="hidden" onChange={handleFileUpload} />
               <Plus className="w-6 h-6" />
            </label>
          </div>
        )}
      </div>

      {/* --- GLOBAL TOAST NOTIFICATION --- */}
      {toastMessage && (
        <div className="fixed bottom-6 left-1/2 -translate-x-1/2 z-[200] bg-neutral-900 text-white px-4 py-2.5 rounded-lg shadow-xl text-sm font-semibold flex items-center gap-2 animate-in slide-in-from-bottom-4 fade-in duration-300">
          <span className="w-2 h-2 rounded-full bg-red-500"></span>
          {toastMessage}
        </div>
      )}

      {/* --- FLOATING KEYBOARD SHORTCUTS BUTTON --- */}
      <button
        onClick={() => setShowShortcuts(true)}
        className="fixed bottom-3 right-3 z-50 p-1 bg-neutral-800 hover:bg-neutral-700 text-neutral-300 hover:text-white rounded-full shadow-lg hover:shadow-xl transition-all border border-neutral-700 hover:scale-110 active:scale-95"
        title="Keyboard Shortcuts"
      >
        <Keyboard className="w-3 h-3" />
      </button>

    </div>
  );
}
